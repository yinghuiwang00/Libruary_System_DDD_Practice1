# 馆藏上下文（Inventory Context）详细设计

## 1. 上下文概述

### 1.1 职责范围
馆藏上下文负责图书馆的实物库存管理，包括：
- 图书副本的创建和追踪
- 多分馆库存同步
- 实时库存状态管理
- 图书位置和状态追踪
- 库存转移和调拨
- 损坏和丢失处理

### 1.2 业务价值
- 确保库存数据的准确性
- 支持多分馆协同运营
- 提供实时的库存可用性信息
- 优化图书分布和利用率
- 支持库存盘点和管理

### 1.3 核心概念
- **CopyInventory（库存）**: 某个图书在特定图书馆的库存聚合
- **BookCopy（图书副本）**: 单个图书的物理副本
- **Location（位置）**: 图书在图书馆中的具体位置
- **Library（图书馆）**: 物理图书馆位置
- **InventoryStatus（库存状态）**: 副本的当前状态

## 2. 领域模型设计

### 2.1 聚合根：CopyInventory

```java
package com.library.inventory.domain.model;

import javax.persistence.*;
import java.time.LocalDateTime;
import java.util.ArrayList;
import java.util.List;
import java.util.Objects;

/**
 * 库存聚合根
 * 管理某个图书在特定图书馆的完整库存
 */
@Entity
@Table(name = "copy_inventories")
public class CopyInventory {
    
    @EmbeddedId
    private CopyInventoryId id;
    
    @ManyToOne(fetch = FetchType.LAZY)
    @JoinColumn(name = "book_id")
    private BookId bookId;
    
    @ManyToOne(fetch = FetchType.LAZY)
    @JoinColumn(name = "library_id")
    private Library library;
    
    @Column(name = "total_copies")
    private Integer totalCopies;
    
    @Column(name = "available_copies")
    private Integer availableCopies;
    
    @OneToMany(mappedBy = "inventory", cascade = CascadeType.ALL, orphanRemoval = true)
    private List<BookCopy> copies = new ArrayList<>();
    
    @Column(name = "last_updated")
    private LocalDateTime lastUpdated;
    
    @Column(name = "updated_by")
    private String updatedBy;
    
    // 构造函数
    protected CopyInventory() {
        // JPA required
    }
    
    public CopyInventory(CopyInventoryId id, BookId bookId, Library library, String createdBy) {
        this.id = Objects.requireNonNull(id, "Inventory ID cannot be null");
        this.bookId = Objects.requireNonNull(bookId, "Book ID cannot be null");
        this.library = Objects.requireNonNull(library, "Library cannot be null");
        this.totalCopies = 0;
        this.availableCopies = 0;
        this.lastUpdated = LocalDateTime.now();
        this.updatedBy = createdBy;
    }
    
    // 领域行为：添加副本
    public BookCopy addCopy(String barcode, Location location, String acquisitionMethod) {
        validateCopyAddition();
        
        CopyId copyId = CopyId.generate();
        BookCopy copy = new BookCopy(copyId, this, barcode, location, acquisitionMethod);
        
        this.copies.add(copy);
        this.totalCopies++;
        this.availableCopies++;
        this.lastUpdated = LocalDateTime.now();
        
        return copy;
    }
    
    // 领域行为：移除副本
    public void removeCopy(CopyId copyId, String reason) {
        BookCopy copy = findCopy(copyId);
        
        if (!copy.canBeRemoved()) {
            throw new InvalidOperationException("Copy cannot be removed in current state: " + copy.getStatus());
        }
        
        copy.markAsRemoved(reason);
        this.copies.remove(copy);
        this.totalCopies--;
        if (copy.isAvailable()) {
            this.availableCopies--;
        }
        this.lastUpdated = LocalDateTime.now();
    }
    
    // 领域行为：获取可用副本
    public BookCopy getAvailableCopy() {
        return this.copies.stream()
            .filter(BookCopy::isAvailable)
            .findFirst()
            .orElseThrow(() -> NoAvailableCopyException.forBook(this.bookId));
    }
    
    // 领域行为：检查是否有可用副本
    public boolean hasAvailableCopies() {
        return this.availableCopies > 0;
    }
    
    // 领域行为：更新副本状态（由副本自身调用）
    public void updateCopyStatus(BookCopy copy, CopyStatus oldStatus, CopyStatus newStatus) {
        if (oldStatus == CopyStatus.AVAILABLE && newStatus != CopyStatus.AVAILABLE) {
            this.availableCopies--;
        } else if (oldStatus != CopyStatus.AVAILABLE && newStatus == CopyStatus.AVAILABLE) {
            this.availableCopies++;
        }
        this.lastUpdated = LocalDateTime.now();
    }
    
    // 领域行为：批量添加副本
    public List<BookCopy> addCopies(int count, Location location, String acquisitionMethod) {
        if (count <= 0) {
            throw new IllegalArgumentException("Copy count must be positive");
        }
        
        List<BookCopy> newCopies = new ArrayList<>();
        for (int i = 0; i < count; i++) {
            String barcode = generateBarcode(this.bookId, this.library, this.totalCopies + i + 1);
            BookCopy copy = addCopy(barcode, location, acquisitionMethod);
            newCopies.add(copy);
        }
        
        return newCopies;
    }
    
    // 领域行为：转移副本到其他库存
    public void transferCopy(CopyId copyId, CopyInventory targetInventory, String reason) {
        BookCopy copy = findCopy(copyId);
        
        if (!copy.canBeTransferred()) {
            throw new InvalidOperationException("Copy cannot be transferred in current state: " + copy.getStatus());
        }
        
        // 从当前库存移除
        this.copies.remove(copy);
        this.totalCopies--;
        if (copy.isAvailable()) {
            this.availableCopies--;
        }
        
        // 添加到目标库存
        copy.transferTo(targetInventory);
        targetInventory.receiveTransfer(copy);
        
        this.lastUpdated = LocalDateTime.now();
    }
    
    // 领域行为：接收转移的副本
    public void receiveTransfer(BookCopy copy) {
        copy.setInventory(this);
        this.copies.add(copy);
        this.totalCopies++;
        if (copy.isAvailable()) {
            this.availableCopies++;
        }
        this.lastUpdated = LocalDateTime.now();
    }
    
    // 领域行为：库存盘点
    public InventoryReconciliationResult reconcile(List<CopyId> physicalCopyIds) {
        List<CopyId> systemCopyIds = this.copies.stream()
            .map(BookCopy::getId)
            .collect(Collectors.toList());
        
        List<CopyId> missingCopies = new ArrayList<>(systemCopyIds);
        missingCopies.removeAll(physicalCopyIds);
        
        List<CopyId> extraCopies = new ArrayList<>(physicalCopyIds);
        extraCopies.removeAll(systemCopyIds);
        
        // 标记丢失的副本
        for (CopyId missingCopyId : missingCopies) {
            BookCopy copy = findCopy(missingCopyId);
            if (copy.isAvailable()) {
                copy.markAsLost("Missing during inventory reconciliation");
                this.availableCopies--;
            }
        }
        
        this.lastUpdated = LocalDateTime.now();
        
        return new InventoryReconciliationResult(missingCopies, extraCopies, LocalDateTime.now());
    }
    
    // 验证方法
    private void validateCopyAddition() {
        if (this.totalCopies >= MAX_COPIES_PER_INVENTORY) {
            throw new InvalidOperationException("Maximum copies per inventory reached: " + MAX_COPIES_PER_INVENTORY);
        }
    }
    
    // 辅助方法
    private BookCopy findCopy(CopyId copyId) {
        return this.copies.stream()
            .filter(copy -> copy.getId().equals(copyId))
            .findFirst()
            .orElseThrow(() -> new CopyNotFoundException(copyId));
    }
    
    private String generateBarcode(BookId bookId, Library library, int sequence) {
        return String.format("%s-%s-%06d", 
            library.getCode(), 
            bookId.getValue().substring(0, 8), 
            sequence);
    }
    
    // 业务查询方法
    public double getAvailabilityRate() {
        if (this.totalCopies == 0) return 0.0;
        return (double) this.availableCopies / this.totalCopies;
    }
    
    public boolean isLowStock() {
        return this.availableCopies <= LOW_STOCK_THRESHOLD;
    }
    
    // Getters
    public CopyInventoryId getId() { return id; }
    public BookId getBookId() { return bookId; }
    public Library getLibrary() { return library; }
    public Integer getTotalCopies() { return totalCopies; }
    public Integer getAvailableCopies() { return availableCopies; }
    public List<BookCopy> getCopies() { return Collections.unmodifiableList(copies); }
    public LocalDateTime getLastUpdated() { return lastUpdated; }
    
    // 常量
    private static final int MAX_COPIES_PER_INVENTORY = 1000;
    private static final int LOW_STOCK_THRESHOLD = 2;
}
```

### 2.2 实体：BookCopy

```java
package com.library.inventory.domain.model;

import javax.persistence.*;
import java.time.LocalDate;
import java.time.LocalDateTime;
import java.util.Objects;

/**
 * 图书副本实体
 * 表示单个图书的物理副本
 */
@Entity
@Table(name = "book_copies")
public class BookCopy {
    
    @EmbeddedId
    private CopyId id;
    
    @ManyToOne(fetch = FetchType.LAZY)
    @JoinColumn(name = "inventory_id")
    private CopyInventory inventory;
    
    @Column(nullable = false, unique = true, length = 50)
    private String barcode;
    
    @Embedded
    private Location location;
    
    @Enumerated(EnumType.STRING)
    @Column(nullable = false)
    private CopyStatus status;
    
    @Column(name = "acquisition_date")
    private LocalDate acquisitionDate;
    
    @Column(name = "acquisition_method", length = 50)
    private String acquisitionMethod;
    
    @Column(name = "acquisition_cost")
    private BigDecimal acquisitionCost;
    
    @Column(name = "condition", length = 20)
    private CopyCondition condition;
    
    @Column(name = "last_borrowed_date")
    private LocalDate lastBorrowedDate;
    
    @Column(name = "borrow_count")
    private Integer borrowCount;
    
    @Column(name = "damaged_date")
    private LocalDate damagedDate;
    
    @Column(name = "damage_description", length = 500)
    private String damageDescription;
    
    @Column(name = "lost_date")
    private LocalDate lostDate;
    
    @Column(name = "lost_reason", length = 200)
    private String lostReason;
    
    @Column(name = "removed_date")
    private LocalDate removedDate;
    
    @Column(name = "removal_reason", length = 200)
    private String removalReason;
    
    @Column(name = "created_at", nullable = false, updatable = false)
    private LocalDateTime createdAt;
    
    @Column(name = "updated_at")
    private LocalDateTime updatedAt;
    
    protected BookCopy() {
        // JPA required
    }
    
    public BookCopy(CopyId id, CopyInventory inventory, String barcode, 
                   Location location, String acquisitionMethod) {
        this.id = Objects.requireNonNull(id, "Copy ID cannot be null");
        this.inventory = Objects.requireNonNull(inventory, "Inventory cannot be null");
        this.barcode = validateBarcode(barcode);
        this.location = Objects.requireNonNull(location, "Location cannot be null");
        this.status = CopyStatus.AVAILABLE;
        this.acquisitionDate = LocalDate.now();
        this.acquisitionMethod = acquisitionMethod;
        this.condition = CopyCondition.NEW;
        this.borrowCount = 0;
        this.createdAt = LocalDateTime.now();
        this.updatedAt = LocalDateTime.now();
    }
    
    // 领域行为：标记为借出
    public void markAsBorrowed() {
        if (this.status != CopyStatus.AVAILABLE) {
            throw new InvalidOperationException("Copy is not available for borrowing: " + this.status);
        }
        
        CopyStatus oldStatus = this.status;
        this.status = CopyStatus.BORROWED;
        this.lastBorrowedDate = LocalDate.now();
        this.borrowCount++;
        this.updatedAt = LocalDateTime.now();
        
        if (this.inventory != null) {
            this.inventory.updateCopyStatus(this, oldStatus, this.status);
        }
    }
    
    // 领域行为：标记为归还
    public void markAsReturned() {
        if (this.status != CopyStatus.BORROWED) {
            throw new InvalidOperationException("Copy is not borrowed: " + this.status);
        }
        
        CopyStatus oldStatus = this.status;
        this.status = CopyStatus.AVAILABLE;
        this.updatedAt = LocalDateTime.now();
        
        if (this.inventory != null) {
            this.inventory.updateCopyStatus(this, oldStatus, this.status);
        }
    }
    
    // 领域行为：标记为预约
    public void markAsReserved() {
        if (this.status != CopyStatus.AVAILABLE) {
            throw new InvalidOperationException("Copy is not available for reservation: " + this.status);
        }
        
        CopyStatus oldStatus = this.status;
        this.status = CopyStatus.RESERVED;
        this.updatedAt = LocalDateTime.now();
        
        if (this.inventory != null) {
            this.inventory.updateCopyStatus(this, oldStatus, this.status);
        }
    }
    
    // 领域行为：标记为损坏
    public void markAsDamaged(String damageDescription) {
        if (this.status == CopyStatus.REMOVED) {
            throw new InvalidOperationException("Cannot mark removed copy as damaged");
        }
        
        CopyStatus oldStatus = this.status;
        this.status = CopyStatus.DAMAGED;
        this.damagedDate = LocalDate.now();
        this.damageDescription = damageDescription;
        this.condition = CopyCondition.DAMAGED;
        this.updatedAt = LocalDateTime.now();
        
        if (this.inventory != null && oldStatus == CopyStatus.AVAILABLE) {
            this.inventory.updateCopyStatus(this, oldStatus, this.status);
        }
    }
    
    // 领域行为：标记为丢失
    public void markAsLost(String reason) {
        if (this.status == CopyStatus.REMOVED) {
            throw new InvalidOperationException("Cannot mark removed copy as lost");
        }
        
        CopyStatus oldStatus = this.status;
        this.status = CopyStatus.LOST;
        this.lostDate = LocalDate.now();
        this.lostReason = reason;
        this.updatedAt = LocalDateTime.now();
        
        if (this.inventory != null && oldStatus == CopyStatus.AVAILABLE) {
            this.inventory.updateCopyStatus(this, oldStatus, this.status);
        }
    }
    
    // 领域行为：标记为维修中
    public void markAsUnderRepair() {
        if (this.status != CopyStatus.DAMAGED) {
            throw new InvalidOperationException("Only damaged copies can be marked as under repair");
        }
        
        this.status = CopyStatus.UNDER_REPAIR;
        this.updatedAt = LocalDateTime.now();
    }
    
    // 领域行为：标记为维修完成
    public void markAsRepaired() {
        if (this.status != CopyStatus.UNDER_REPAIR) {
            throw new InvalidOperationException("Copy is not under repair");
        }
        
        CopyStatus oldStatus = this.status;
        this.status = CopyStatus.AVAILABLE;
        this.condition = CopyCondition.GOOD;
        this.updatedAt = LocalDateTime.now();
        
        if (this.inventory != null) {
            this.inventory.updateCopyStatus(this, oldStatus, this.status);
        }
    }
    
    // 领域行为：标记为移除
    public void markAsRemoved(String reason) {
        if (this.status == CopyStatus.BORROWED) {
            throw new InvalidOperationException("Cannot remove borrowed copy");
        }
        
        CopyStatus oldStatus = this.status;
        this.status = CopyStatus.REMOVED;
        this.removedDate = LocalDate.now();
        this.removalReason = reason;
        this.updatedAt = LocalDateTime.now();
        
        if (this.inventory != null && oldStatus == CopyStatus.AVAILABLE) {
            this.inventory.updateCopyStatus(this, oldStatus, this.status);
        }
    }
    
    // 领域行为：更新位置
    public void updateLocation(Location newLocation) {
        this.location = Objects.requireNonNull(newLocation, "Location cannot be null");
        this.updatedAt = LocalDateTime.now();
    }
    
    // 领域行为：更新状况
    public void updateCondition(CopyCondition newCondition) {
        this.condition = Objects.requireNonNull(newCondition, "Condition cannot be null");
        this.updatedAt = LocalDateTime.now();
    }
    
    // 领域行为：转移库存
    public void transferTo(CopyInventory targetInventory) {
        if (this.status == CopyStatus.BORROWED) {
            throw new InvalidOperationException("Cannot transfer borrowed copy");
        }
        if (this.status == CopyStatus.REMOVED) {
            throw new InvalidOperationException("Cannot transfer removed copy");
        }
        
        this.inventory = Objects.requireNonNull(targetInventory, "Target inventory cannot be null");
        this.updatedAt = LocalDateTime.now();
    }
    
    // 验证方法
    private String validateBarcode(String barcode) {
        if (barcode == null || barcode.trim().isEmpty()) {
            throw new IllegalArgumentException("Barcode cannot be empty");
        }
        if (barcode.length() > 50) {
            throw new IllegalArgumentException("Barcode cannot exceed 50 characters");
        }
        return barcode.trim();
    }
    
    // 业务查询方法
    public boolean isAvailable() {
        return this.status == CopyStatus.AVAILABLE;
    }
    
    public boolean isBorrowed() {
        return this.status == CopyStatus.BORROWED;
    }
    
    public boolean canBeRemoved() {
        return this.status != CopyStatus.BORROWED && this.status != CopyStatus.REMOVED;
    }
    
    public boolean canBeTransferred() {
        return this.status != CopyStatus.BORROWED && this.status != CopyStatus.REMOVED;
    }
    
    public boolean isDamagedOrLost() {
        return this.status == CopyStatus.DAMAGED || 
               this.status == CopyStatus.LOST ||
               this.status == CopyStatus.UNDER_REPAIR;
    }
    
    // Getters
    public CopyId getId() { return id; }
    public CopyInventory getInventory() { return inventory; }
    public String getBarcode() { return barcode; }
    public Location getLocation() { return location; }
    public CopyStatus getStatus() { return status; }
    public LocalDate getAcquisitionDate() { return acquisitionDate; }
    public String getAcquisitionMethod() { return acquisitionMethod; }
    public BigDecimal getAcquisitionCost() { return acquisitionCost; }
    public CopyCondition getCondition() { return condition; }
    public LocalDate getLastBorrowedDate() { return lastBorrowedDate; }
    public Integer getBorrowCount() { return borrowCount; }
}
```

### 2.3 值对象：Location

```java
package com.library.inventory.domain.model;

import javax.persistence.Embeddable;
import java.util.Objects;
import java.util.regex.Pattern;

/**
 * 位置值对象
 * 表示图书在图书馆中的具体位置
 */
@Embeddable
public class Location {
    
    private static final Pattern LOCATION_CODE_PATTERN = Pattern.compile("^[A-Z]{1,2}\\d{1,3}-[A-Z]\\d{2}-\\d{3}$");
    
    @Column(name = "library_code", nullable = false, length = 10)
    private String libraryCode;
    
    @Column(name = "floor", nullable = false)
    private Integer floor;
    
    @Column(name = "zone", length = 10)
    private String zone;
    
    @Column(name = "aisle", length = 10)
    private String aisle;
    
    @Column(name = "shelf", length = 10)
    private String shelf;
    
    @Column(name = "position", length = 10)
    private String position;
    
    @Column(name = "location_code", length = 50)
    private String locationCode;
    
    protected Location() {
        // JPA required
    }
    
    public Location(String libraryCode, Integer floor, String zone, String aisle, String shelf, String position) {
        this.libraryCode = validateLibraryCode(libraryCode);
        this.floor = validateFloor(floor);
        this.zone = zone;
        this.aisle = aisle;
        this.shelf = shelf;
        this.position = position;
        this.locationCode = generateLocationCode();
    }
    
    // 简化构造函数
    public Location(String libraryCode, String locationCode) {
        this.libraryCode = validateLibraryCode(libraryCode);
        this.locationCode = validateLocationCode(locationCode);
        parseLocationCode(locationCode);
    }
    
    private String validateLibraryCode(String code) {
        if (code == null || code.trim().isEmpty()) {
            throw new IllegalArgumentException("Library code cannot be empty");
        }
        return code.toUpperCase().trim();
    }
    
    private Integer validateFloor(Integer floor) {
        if (floor == null || floor < 0) {
            throw new IllegalArgumentException("Floor must be non-negative");
        }
        return floor;
    }
    
    private String validateLocationCode(String code) {
        if (code == null || code.trim().isEmpty()) {
            throw new IllegalArgumentException("Location code cannot be empty");
        }
        if (!LOCATION_CODE_PATTERN.matcher(code).matches()) {
            throw new IllegalArgumentException("Invalid location code format. Expected: A123-B01-001");
        }
        return code.toUpperCase().trim();
    }
    
    private String generateLocationCode() {
        String zoneStr = (zone != null && !zone.isEmpty()) ? zone : "A";
        String aisleStr = (aisle != null && !aisle.isEmpty()) ? aisle : "01";
        String positionStr = (position != null && !position.isEmpty()) ? position : "001";
        
        return String.format("%s%s-%s-%03d", 
            zoneStr, aisleStr, shelf != null ? shelf : "A", 
            Integer.parseInt(positionStr));
    }
    
    private void parseLocationCode(String code) {
        // 解析位置代码: A123-B01-001
        String[] parts = code.split("-");
        if (parts.length == 3) {
            String zoneAisle = parts[0];
            this.zone = zoneAisle.substring(0, 1);
            this.aisle = zoneAisle.substring(1);
            this.shelf = parts[1];
            this.position = parts[2];
        }
    }
    
    public String getFullDescription() {
        return String.format("Library %s, Floor %d, Zone %s, Aisle %s, Shelf %s, Position %s",
            libraryCode, floor, zone, aisle, shelf, position);
    }
    
    public boolean isSameFloor(Location other) {
        return this.floor.equals(other.floor);
    }
    
    public boolean isSameZone(Location other) {
        return this.zone != null && this.zone.equals(other.zone);
    }
    
    // Getters
    public String getLibraryCode() { return libraryCode; }
    public Integer getFloor() { return floor; }
    public String getZone() { return zone; }
    public String getAisle() { return aisle; }
    public String getShelf() { return shelf; }
    public String getPosition() { return position; }
    public String getLocationCode() { return locationCode; }
    
    @Override
    public boolean equals(Object o) {
        if (this == o) return true;
        if (o == null || getClass() != o.getClass()) return false;
        Location location = (Location) o;
        return locationCode.equals(location.locationCode);
    }
    
    @Override
    public int hashCode() {
        return Objects.hash(locationCode);
    }
    
    @Override
    public String toString() {
        return locationCode;
    }
}
```

### 2.6 实体：Library

```java
package com.library.inventory.domain.model;

import javax.persistence.*;
import java.time.LocalDateTime;
import java.util.ArrayList;
import java.util.List;
import java.util.Objects;

/**
 * 图书馆实体
 * 表示物理图书馆位置
 */
@Entity
@Table(name = "libraries")
public class Library {
    
    @EmbeddedId
    private LibraryId id;
    
    @Column(nullable = false, unique = true, length = 20)
    private String code;
    
    @Column(nullable = false, length = 200)
    private String name;
    
    @Column(length = 500)
    private String address;
    
    @Column(length = 100)
    private String city;
    
    @Column(length = 100)
    private String province;
    
    @Column(length = 20)
    private String postalCode;
    
    @Column(name = "phone", length = 20)
    private String phone;
    
    @Column(name = "email", length = 100)
    private String email;
    
    @Column(name = "opening_hours", length = 200)
    private String openingHours;
    
    @Column(name = "total_floors")
    private Integer totalFloors;
    
    @Column(name = "is_active")
    private Boolean active;
```

```java
package com.library.inventory.domain.service;

import com.library.inventory.domain.model.*;

import com.library.shared.domain.event.DomainEventPublisher;
import org.springframework.stereotype.Service;

import java.math.BigDecimal;

/**
 * 库存管理领域服务
 * 处理跨聚合的复杂业务逻辑
 */
@Service
public class InventoryManagementService {
    
    private final CopyInventoryRepository inventoryRepository;
    private final BookCopyRepository copyRepository;
    private final LibraryRepository libraryRepository;
    private final DomainEventPublisher eventPublisher;
    private final InventorySyncService inventorySyncService;
    
    public InventoryManagementService(CopyInventoryRepository inventoryRepository,
                                      BookCopyRepository copyRepository,
                                      LibraryRepository libraryRepository,
                                      DomainEventPublisher eventPublisher,
                                      InventorySyncService inventorySyncService) {
        this.inventoryRepository = inventoryRepository;
        this.copyRepository = copyRepository;
        this.libraryRepository = libraryRepository;
        this.eventPublisher = eventPublisher;
        this.inventorySyncService = inventorySyncService;
    }
    
    /**
     * 为图书在指定图书馆创建初始库存
     */
    @Transactional
    public CopyInventory createInitialInventory(BookId bookId, LibraryId libraryId, 
                                                int initialCopyCount, Location defaultLocation,
                                                String createdBy) {
        // 1. 验证图书馆存在且活跃
        Library library = libraryRepository.findById(libraryId)
            .orElseThrow(() -> new LibraryNotFoundException(libraryId));
        
        if (!library.isActive()) {
            throw new InvalidOperationException("Library is not active: " + libraryId);
        }
        
        // 2. 检查是否已存在库存
        if (inventoryRepository.existsByBookAndLibrary(bookId, libraryId)) {
            throw new DuplicateInventoryException(bookId, libraryId);
        }
        
        // 3. 创建库存聚合
        CopyInventoryId inventoryId = CopyInventoryId.generate();
        CopyInventory inventory = new CopyInventory(inventoryId, bookId, library, createdBy);
        
        // 4. 添加初始副本
        if (initialCopyCount > 0) {
            List<BookCopy> copies = inventory.addCopies(initialCopyCount, defaultLocation, "PURCHASE");
            
            // 设置采购成本（如果有）
            for (BookCopy copy : copies) {
                copy.setAcquisitionCost(calculateDefaultAcquisitionCost());
            }
        }
        
        // 5. 保存库存
        CopyInventory savedInventory = inventoryRepository.save(inventory);
        
        // 6. 发布领域事件
        eventPublisher.publish(new InventoryCreatedEvent(
            savedInventory.getId(),
            bookId,
            libraryId,
            initialCopyCount,
            LocalDateTime.now()
        ));
        
        return savedInventory;
    }
    
    /**
     * 添加副本到现有库存
     */
    @Transactional
    public BookCopy addCopyToInventory(CopyInventoryId inventoryId, Location location, 
                                       String acquisitionMethod, BigDecimal cost, String addedBy) {
        CopyInventory inventory = inventoryRepository.findById(inventoryId)
            .orElseThrow(() -> new InventoryNotFoundException(inventoryId));
        
        String barcode = generateUniqueBarcode(inventory);
        BookCopy copy = inventory.addCopy(barcode, location, acquisitionMethod);
        
        if (cost != null) {
            copy.setAcquisitionCost(cost);
        }
        
        inventoryRepository.save(inventory);
        
        eventPublisher.publish(new CopyAddedEvent(
            copy.getId(),
            inventoryId,
            inventory.getBookId(),
            barcode,
            LocalDateTime.now()
        ));
        
        return copy;
    }
    
    /**
     * 批量添加副本
     */
    @Transactional
    public List<BookCopy> batchAddCopies(CopyInventoryId inventoryId, int count, 
                                        Location location, String acquisitionMethod, 
                                        BigDecimal cost, String addedBy) {
        CopyInventory inventory = inventoryRepository.findById(inventoryId)
            .orElseThrow(() -> new InventoryNotFoundException(inventoryId));
        
        List<BookCopy> copies = inventory.addCopies(count, location, acquisitionMethod);
        
        if (cost != null) {
            for (BookCopy copy : copies) {
                copy.setAcquisitionCost(cost);
            }
        }
        
        inventoryRepository.save(inventory);
        
        eventPublisher.publish(new CopiesBatchAddedEvent(
            inventoryId,
            inventory.getBookId(),
            count,
            LocalDateTime.now()
        ));
        
        return copies;
    }
    
    /**
     * 转移副本到其他图书馆
     */
    @Transactional
    public void transferCopyBetweenLibraries(CopyId copyId, LibraryId targetLibraryId, 
                                            String reason, String transferredBy) {
        // 1. 获取源副本和库存
        BookCopy copy = copyRepository.findById(copyId)
            .orElseThrow(() -> new CopyNotFoundException(copyId));
        
        CopyInventory sourceInventory = copy.getInventory();
        
        // 2. 获取或创建目标库存
        CopyInventory targetInventory = inventoryRepository
            .findByBookAndLibrary(sourceInventory.getBookId(), targetLibraryId)
            .orElseGet(() -> {
                Library targetLibrary = libraryRepository.findById(targetLibraryId)
                    .orElseThrow(() -> new LibraryNotFoundException(targetLibraryId));
                
                return createInitialInventory(
                    sourceInventory.getBookId(),
                    targetLibraryId,
                    0,
                    copy.getLocation(), // 使用原位置作为默认位置
                    transferredBy
                );
            });
        
        // 3. 执行转移
        sourceInventory.transferCopy(copyId, targetInventory, reason);
        
        // 4. 保存更新
        inventoryRepository.save(sourceInventory);
        inventoryRepository.save(targetInventory);
        
        // 5. 发布事件
        eventPublisher.publish(new CopyTransferredEvent(
            copyId,
            sourceInventory.getId(),
            targetInventory.getId(),
            sourceInventory.getLibrary().getId(),
            targetLibraryId,
            reason,
            LocalDateTime.now()
        ));
        
        // 6. 同步到其他系统
        inventorySyncService.syncCopyTransfer(copy, sourceInventory.getLibrary(), targetInventory.getLibrary());
    }
    
    /**
     * 处理副本损坏
     */
    @Transactional
    public void reportCopyDamage(CopyId copyId, String damageDescription, String reportedBy) {
        BookCopy copy = copyRepository.findById(copyId)
            .orElseThrow(() -> new CopyNotFoundException(copyId));
        
        copy.markAsDamaged(damageDescription);
        copyRepository.save(copy);
        
        eventPublisher.publish(new CopyDamagedEvent(
            copyId,
            copy.getInventory().getId(),
            damageDescription,
            LocalDateTime.now()
        ));
        
        // 检查是否需要补充库存
        checkAndReplenishInventory(copy.getInventory());
    }
    
    /**
     * 处理副本丢失
     */
    @Transactional
    public void reportCopyLoss(CopyId copyId, String reason, String reportedBy) {
        BookCopy copy = copyRepository.findById(copyId)
            .orElseThrow(() -> new CopyNotFoundException(copyId));
        
        copy.markAsLost(reason);
        copyRepository.save(copy);
        
        eventPublisher.publish(new CopyLostEvent(
            copyId,
            copy.getInventory().getId(),
            reason,
            LocalDateTime.now()
        ));
        
        // 检查是否需要补充库存
        checkAndReplenishInventory(copy.getInventory());
    }
    
    /**
     * 执行库存盘点
     */
    @Transactional
    public InventoryReconciliationResult performInventoryReconciliation(
            CopyInventoryId inventoryId, List<CopyId> physicalCopyIds, String performedBy) {
        
        CopyInventory inventory = inventoryRepository.findById(inventoryId)
            .orElseThrow(() -> new InventoryNotFoundException(inventoryId));
        
        InventoryReconciliationResult result = inventory.reconcile(physicalCopyIds);
        inventoryRepository.save(inventory);
        
        // 如果有丢失的副本，发布事件
        if (!result.getMissingCopies().isEmpty()) {
            eventPublisher.publish(new CopiesLostDuringReconciliationEvent(
                inventoryId,
                result.getMissingCopies(),
                performedBy,
                result.getReconciliationDate()
            ));
        }
        
        // 如果有多余的副本，需要进一步调查
        if (!result.getExtraCopies().isEmpty()) {
            eventPublisher.publish(new ExtraCopiesFoundEvent(
                inventoryId,
                result.getExtraCopies(),
                performedBy,
                result.getReconciliationDate()
            ));
        }
        
        return result;
    }
    
    /**
     * 获取图书在所有图书馆的库存概览
     */
    @Transactional(readOnly = true)
    public List<InventorySummary> getInventoryOverview(BookId bookId) {
        List<CopyInventory> inventories = inventoryRepository.findByBookId(bookId);
        
        return inventories.stream()
            .map(inventory -> new InventorySummary(
                inventory.getId(),
                inventory.getLibrary().getId(),
                inventory.getLibrary().getName(),
                inventory.getTotalCopies(),
                inventory.getAvailableCopies(),
                inventory.getAvailabilityRate(),
                inventory.isLowStock()
            ))
            .collect(Collectors.toList());
    }
    
    /**
     * 查找最近的可用副本
     */
    @Transactional(readOnly = true)
    public AvailableCopyLocation findNearestAvailableCopy(BookId bookId, LibraryId preferredLibraryId) {
        // 优先查找首选图书馆
        CopyInventory preferredInventory = inventoryRepository
            .findByBookAndLibrary(bookId, preferredLibraryId)
            .orElse(null);
        
        if (preferredInventory != null && preferredInventory.hasAvailableCopies()) {
            BookCopy copy = preferredInventory.getAvailableCopy();
            return new AvailableCopyLocation(copy.getId(), preferredLibraryId, copy.getLocation(), 0);
        }
        
        // 查找其他有库存的图书馆
        List<CopyInventory> availableInventories = inventoryRepository
            .findByBookIdAndAvailableCopiesGreaterThan(bookId, 0);
        
        // 这里可以添加距离计算逻辑，暂时返回第一个可用的
        if (!availableInventories.isEmpty()) {
            CopyInventory inventory = availableInventories.get(0);
            BookCopy copy = inventory.getAvailableCopy();
            return new AvailableCopyLocation(copy.getId(), inventory.getLibrary().getId(), copy.getLocation(), 1);
        }
        
        throw new NoAvailableCopyException(bookId);
    }
    
    // 辅助方法
    private String generateUniqueBarcode(CopyInventory inventory) {
        return String.format("%s-%s-%06d", 
            inventory.getLibrary().getCode(),
            inventory.getBookId().getValue().substring(0, 8),
            inventory.getTotalCopies() + 1);
    }
    
    private BigDecimal calculateDefaultAcquisitionCost() {
        // 这里可以实现更复杂的成本计算逻辑
        return new BigDecimal("25.00"); // 默认成本
    }
    
    private void checkAndReplenishInventory(CopyInventory inventory) {
        if (inventory.isLowStock()) {
            eventPublisher.publish(new LowStockAlertEvent(
                inventory.getId(),
                inventory.getBookId(),
                inventory.getAvailableCopies(),
                LocalDateTime.now()
            ));
        }
    }
```

```java
                    inventory.getBookId(),
                    inventory.getLibrary().getId(),
                    inventory.getTotalCopies(),
                    inventory.getAvailableCopies()

            } catch (Exception e) {

            }
        });
    }

            }
        });
```

```java
        request.setCopyId(CopyId.of(copyId));
        inventoryApplicationService.reportDamage(request);
        return ResponseEntity.ok(ApiResponse.success(null));
    }
    
    /**
     * 获取库存概览
     * GET /api/inventory/books/{bookId}/overview
     */
    @GetMapping("/books/{bookId}/overview")
    public ResponseEntity<ApiResponse<List<InventorySummaryDTO>>> getInventoryOverview(
            @PathVariable String bookId) {
        List<InventorySummaryDTO> overview = inventoryApplicationService
            .getInventoryOverview(BookId.of(bookId));
        return ResponseEntity.ok(ApiResponse.success(overview));
    }
    
    /**
     * 库存盘点
     * POST /api/inventory/inventories/{inventoryId}/reconcile
     */
    @PostMapping("/inventories/{inventoryId}/reconcile")
    public ResponseEntity<ApiResponse<ReconciliationResultDTO>> performReconciliation(
            @PathVariable String inventoryId,
            @Valid @RequestBody PerformReconciliationRequest request) {
        request.setInventoryId(CopyInventoryId.of(inventoryId));
        ReconciliationResultDTO result = inventoryApplicationService.performReconciliation(request);
        return ResponseEntity.ok(ApiResponse.success(result));
    }
}
```
