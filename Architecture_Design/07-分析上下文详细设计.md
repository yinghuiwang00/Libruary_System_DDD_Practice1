# 分析上下文（Analytics Context）详细设计

## 1. 上下文概述

### 1.1 职责范围
分析上下文负责图书馆系统的数据分析和报表生成，包括：
- 借阅统计分析
- 热门图书分析
- 读者行为分析
- 库存利用率分析
- 财务数据分析
- 趋势预测分析
- 实时数据仪表板
- 定制化报表生成

### 1.2 业务价值
- 为图书馆管理提供数据支持
- 优化图书采购和库存管理
- 了解读者需求和行为
- 提高运营效率和服务质量
- 支持战略决策制定

### 1.3 核心概念
- **AnalyticsReport（分析报表）**: 各种统计报表
- **Metric（指标）**: 业务指标定义
- **Dashboard（仪表板）**: 数据可视化面板
- **DataCube（数据立方体）**: 多维数据分析
- **TrendAnalysis（趋势分析）**: 时间序列分析
- **PredictionModel（预测模型）**: 预测分析模型

package com.library.analytics.domain.model;

import javax.persistence.*;
import java.time.LocalDateTime;
import java.time.LocalDate;
import java.util.ArrayList;
import java.util.List;
import java.util.Objects;

/**
 * 分析报表聚合根
 * 管理各类统计报表的生成和存储
 */
@Entity
@Table(name = "analytics_reports", indexes = {
    @Index(name = "idx_report_type", columnList = "report_type"),
    @Index(name = "idx_report_period", columnList = "report_period"),
    @Index(name = "idx_report_date", columnList = "report_date")

public class AnalyticsReport {
    
    @EmbeddedId
    private ReportId id;
    
    @Enumerated(EnumType.STRING)
    @Column(nullable = false)
    private ReportType reportType;
    
    @Enumerated(EnumType.STRING)
    @Column(nullable = false)
    private ReportPeriod reportPeriod;
    
    @Column(name = "report_date", nullable = false)
    private LocalDate reportDate;
    
    @Column(name = "start_date")
    private LocalDate startDate;
    
    @Column(name = "end_date")
    private LocalDate endDate;
    
    @Enumerated(EnumType.STRING)
    @Column(nullable = false)
    private ReportStatus status;
    
    @Column(name = "generated_at")
    private LocalDateTime generatedAt;
    
    @Column(name = "generated_by")
    private String generatedBy;
    
    @Column(name = "total_records")
    private Long totalRecords;
    
    @Column(name = "data_summary", columnDefinition = "TEXT")
    private String dataSummary;
    
    @Column(name = "report_data", columnDefinition = "TEXT")
    private String reportData;
    
    @OneToMany(mappedBy = "report", cascade = CascadeType.ALL)
    private List<ReportMetric> metrics = new ArrayList<>();
    
    @OneToMany(mappedBy = "report", cascade = CascadeType.ALL)
    private List<ReportDimension> dimensions = new ArrayList<>();
    
    @Column(name = "created_at", nullable = false, updatable = false)
    private LocalDateTime createdAt;
    
    @Column(name = "updated_at")
    private LocalDateTime updatedAt;
    
    protected AnalyticsReport() {
        // JPA required
    }
    
    public AnalyticsReport(ReportId id, ReportType reportType, ReportPeriod reportPeriod,
                         LocalDate reportDate, LocalDate startDate, LocalDate endDate, String generatedBy) {
        this.id = Objects.requireNonNull(id, "Report ID cannot be null");
        this.reportType = Objects.requireNonNull(reportType, "Report type cannot be null");
        this.reportPeriod = Objects.requireNonNull(reportPeriod, "Report period cannot be null");
        this.reportDate = Objects.requireNonNull(reportDate, "Report date cannot be null");
        this.startDate = startDate;
        this.endDate = endDate;
        this.status = ReportStatus.GENERATING;
        this.generatedBy = generatedBy;
        this.createdAt = LocalDateTime.now();
        this.updatedAt = LocalDateTime.now();
    }
    
    // 领域行为：完成报表生成
    public void complete(Long totalRecords, String dataSummary, String reportData) {
        if (this.status != ReportStatus.GENERATING) {
            throw new InvalidOperationException("Report is not in generating status");
        }
        
        this.status = ReportStatus.COMPLETED;
        this.generatedAt = LocalDateTime.now();
        this.totalRecords = totalRecords;
        this.dataSummary = dataSummary;
        this.reportData = reportData;
        this.updatedAt = LocalDateTime.now();
        
        DomainEventPublisher.publish(new ReportCompletedEvent(
            this.id,
            this.reportType,
            this.reportDate,
            totalRecords,
            LocalDateTime.now()
        ));
    }
    
    // 领域行为：报表生成失败
    public void fail(String errorMessage) {
        if (this.status != ReportStatus.GENERATING) {
            throw new InvalidOperationException("Report is not in generating status");
        }
        
        this.status = ReportStatus.FAILED;
        this.dataSummary = errorMessage;
        this.updatedAt = LocalDateTime.now();
        
        DomainEventPublisher.publish(new ReportFailedEvent(
            this.id,
            this.reportType,
            this.reportDate,
            errorMessage,
            LocalDateTime.now()
        ));
    }
    
    // 领域行为：添加指标
    public void addMetric(String metricName, String metricValue, String metricUnit) {
        ReportMetric metric = new ReportMetric(this, metricName, metricValue, metricUnit);
        this.metrics.add(metric);
        this.updatedAt = LocalDateTime.now();
    }
    
    // 领域行为：添加维度
    public void addDimension(String dimensionName, String dimensionValue) {
        ReportDimension dimension = new ReportDimension(this, dimensionName, dimensionValue);
        this.dimensions.add(dimension);
        this.updatedAt = LocalDateTime.now();
    }
    
    // 领域行为：更新报表数据
    public void updateData(String newData) {
        if (this.status != ReportStatus.COMPLETED) {
            throw new InvalidOperationException("Can only update completed reports");
        }
        
        this.reportData = newData;
        this.updatedAt = LocalDateTime.now();
    }
    
    // 领域行为：重新生成报表
    public void regenerate(String regeneratedBy) {
        this.status = ReportStatus.GENERATING;
        this.generatedBy = regeneratedBy;
        this.totalRecords = null;
        this.dataSummary = null;
        this.reportData = null;
        this.metrics.clear();
        this.dimensions.clear();
        this.updatedAt = LocalDateTime.now();
        
        DomainEventPublisher.publish(new ReportRegeneratedEvent(
            this.id,
            this.reportType,
            LocalDateTime.now()
        ));
    }
    
    // 业务查询方法
    public boolean isCompleted() {
        return this.status == ReportStatus.COMPLETED;
    }
    
    public boolean isGenerating() {
        return this.status == ReportStatus.GENERATING;
    }
    
    public boolean isFailed() {
        return this.status == ReportStatus.FAILED;
    }
    
    public long getProcessingTimeSeconds() {
        if (this.generatedAt == null || this.createdAt == null) {
            return 0;
        }
        return java.time.Duration.between(this.createdAt, this.generatedAt).getSeconds();
    }
    
    // Getters
    public ReportId getId() { return id; }
    public ReportType getReportType() { return reportType; }
    public ReportPeriod getReportPeriod() { return reportPeriod; }
    public LocalDate getReportDate() { return reportDate; }
    public LocalDate getStartDate() { return startDate; }
    public LocalDate getEndDate() { return endDate; }
    public ReportStatus getStatus() { return status; }
    public LocalDateTime getGeneratedAt() { return generatedAt; }
    public String getGeneratedBy() { return generatedBy; }
    public Long getTotalRecords() { return totalRecords; }
    public String getDataSummary() { return dataSummary; }
    public String getReportData() { return reportData; }
    public List<ReportMetric> getMetrics() { return new ArrayList<>(metrics); }
    public List<ReportDimension> getDimensions() { return new ArrayList<>(dimensions); }
}

        this.dashboardType = Objects.requireNonNull(dashboardType, "Dashboard type cannot be null");
        this.ownerId = ownerId;
        this.isPublic = false;
        this.refreshIntervalMinutes = 30; // 默认30分钟刷新
        this.createdAt = LocalDateTime.now();
        this.updatedAt = LocalDateTime.now();
    }
    
    // 领域行为：添加小组件
    public void addWidget(String widgetType, String widgetConfig, int positionX, 
                        int positionY, int width, int height) {
        DashboardWidgetId widgetId = DashboardWidgetId.generate();
        DashboardWidget widget = new DashboardWidget(
            widgetId,
            this,
            widgetType,
            widgetConfig,
            positionX,
            positionY,
            width,
            height
        );
        
        this.widgets.add(widget);
        this.updatedAt = LocalDateTime.now();
    }
    
    // 领域行为：移除小组件
    public void removeWidget(DashboardWidgetId widgetId) {
        boolean removed = this.widgets.removeIf(w -> w.getId().equals(widgetId));
        if (removed) {
            this.updatedAt = LocalDateTime.now();
        }
    }
    
    // 领域行为：更新布局
    public void updateLayout(String layoutConfig) {
        this.layoutConfig = layoutConfig;
        this.updatedAt = LocalDateTime.now();
    }
    
    // 领域行为：设置刷新间隔
    public void setRefreshInterval(Integer minutes) {
        if (minutes == null || minutes < 1) {
            throw new IllegalArgumentException("Refresh interval must be at least 1 minute");
        }
        this.refreshIntervalMinutes = minutes;
        this.updatedAt = LocalDateTime.now();
    }
    
    // 领域行为：标记为已刷新
    public void markAsRefreshed() {
        this.lastRefreshedAt = LocalDateTime.now();
        this.updatedAt = LocalDateTime.now();
    }
    
    // 领域行为：公开/私有化
    public void setPublic(boolean isPublic) {
        this.isPublic = isPublic;
        this.updatedAt = LocalDateTime.now();
    }
    
    // 业务查询方法
    public boolean needsRefresh() {
        if (this.lastRefreshedAt == null) {
            return true;
        }
        LocalDateTime nextRefresh = this.lastRefreshedAt.plusMinutes(this.refreshIntervalMinutes);
        return LocalDateTime.now().isAfter(nextRefresh);
    }
    
    public long getMinutesSinceLastRefresh() {
        if (this.lastRefreshedAt == null) {
            return Long.MAX_VALUE;
        }
        return java.time.Duration.between(this.lastRefreshedAt, LocalDateTime.now()).toMinutes();
    }
    
    // Getters
    public DashboardId getId() { return id; }
    public String getName() { return name; }
    public String getDescription() { return description; }
    public DashboardType getDashboardType() { return dashboardType; }
    public Boolean getIsPublic() { return isPublic; }
    public String getOwnerId() { return ownerId; }
    public String getLayoutConfig() { return layoutConfig; }
    public Integer getRefreshIntervalMinutes() { return refreshIntervalMinutes; }
    public LocalDateTime getLastRefreshedAt() { return lastRefreshedAt; }
    public List<DashboardWidget> getWidgets() { return new ArrayList<>(widgets); }
}

package com.library.analytics.domain.model;

import javax.persistence.Embeddable;
import java.math.BigDecimal;
import java.math.RoundingMode;
import java.time.LocalDate;
import java.util.ArrayList;
import java.util.List;
import java.util.Objects;

/**
 * 趋势分析值对象
 * 存储时间序列分析结果
 */
@Embeddable
public class TrendAnalysis {
    
    @Column(name = "analysis_type", length = 50)
    private String analysisType;
    
    @Column(name = "trend_direction", length = 20)
    private String trendDirection;
    
    @Column(name = "current_value", precision = 15, scale = 2)
    private BigDecimal currentValue;
    
    @Column(name = "previous_value", precision = 15, scale = 2)
    private BigDecimal previousValue;
    
    @Column(name = "change_percentage", precision = 10, scale = 2)
    private BigDecimal changePercentage;
    
    @Column(name = "trend_strength", length = 20)
    private String trendStrength;
    
    @Column(name = "confidence_level", precision = 5, scale = 2)
    private BigDecimal confidenceLevel;
    
    @Column(name = "forecast_value", precision = 15, scale = 2)
    private BigDecimal forecastValue;
    
    @Column(name = "forecast_date")
    private LocalDate forecastDate;
    
    protected TrendAnalysis() {
        // JPA required
    }
    
    public TrendAnalysis(String analysisType, BigDecimal currentValue, BigDecimal previousValue) {
        this.analysisType = Objects.requireNonNull(analysisType, "Analysis type cannot be null");
        this.currentValue = Objects.requireNonNull(currentValue, "Current value cannot be null");
        this.previousValue = previousValue;
        this.calculateTrend();
    }
    
    private void calculateTrend() {
        if (this.previousValue == null || this.previousValue.compareTo(BigDecimal.ZERO) == 0) {
            this.trendDirection = "UNKNOWN";
            this.changePercentage = BigDecimal.ZERO;
            return;
        }
        
        BigDecimal change = this.currentValue.subtract(this.previousValue);
        this.changePercentage = change.divide(this.previousValue, 4, RoundingMode.HALF_UP)
            .multiply(new BigDecimal("100"))
            .setScale(2, RoundingMode.HALF_UP);
        
        if (this.changePercentage.compareTo(BigDecimal.ZERO) > 0) {
            this.trendDirection = "UP";
        } else if (this.changePercentage.compareTo(BigDecimal.ZERO) < 0) {
            this.trendDirection = "DOWN";
        } else {
            this.trendDirection = "STABLE";
        }
        
        // 确定趋势强度
        double absChange = this.changePercentage.abs().doubleValue();
        if (absChange >= 20) {
            this.trendStrength = "STRONG";
        } else if (absChange >= 10) {
            this.trendStrength = "MODERATE";
        } else if (absChange >= 5) {
            this.trendStrength = "WEAK";
        } else {
            this.trendStrength = "NEGLIGIBLE";
        }
    }
    
    public void setForecast(BigDecimal forecastValue, LocalDate forecastDate, BigDecimal confidenceLevel) {
        this.forecastValue = forecastValue;
        this.forecastDate = forecastDate;
        this.confidenceLevel = confidenceLevel;
    }
    
    // 业务查询方法
    public boolean isUpwardTrend() {
        return "UP".equals(this.trendDirection);
    }
    
    public boolean isDownwardTrend() {
        return "DOWN".equals(this.trendDirection);
    }
    
    public boolean isSignificant() {
        return "STRONG".equals(this.trendStrength) || "MODERATE".equals(this.trendStrength);
    }
    
    // Getters
    public String getAnalysisType() { return analysisType; }
    public String getTrendDirection() { return trendDirection; }
    public BigDecimal getCurrentValue() { return currentValue; }
    public BigDecimal getPreviousValue() { return previousValue; }
    public BigDecimal getChangePercentage() { return changePercentage; }
    public String getTrendStrength() { return trendStrength; }
    public BigDecimal getConfidenceLevel() { return confidenceLevel; }
    public BigDecimal getForecastValue() { return forecastValue; }
    public LocalDate getForecastDate() { return forecastDate; }
}

package com.library.analytics.domain.service;

import org.springframework.stereotype.Service;
import org.springframework.transaction.annotation.Transactional;

import java.time.LocalDate;
import java.time.LocalDateTime;
import java.util.List;
import java.util.Map;
import java.util.concurrent.CompletableFuture;

/**
 * 报表生成领域服务
 */
@Service
public class ReportGenerationService {
    
    private final AnalyticsReportRepository reportRepository;
    private final CirculationDataRepository circulationDataRepository;
    private final InventoryDataRepository inventoryDataRepository;
    private final PatronDataRepository patronDataRepository;
    private final FinancialDataRepository financialDataRepository;
    private final DomainEventPublisher eventPublisher;
    
    public ReportGenerationService(AnalyticsReportRepository reportRepository,
                                  CirculationDataRepository circulationDataRepository,
                                  InventoryDataRepository inventoryDataRepository,
                                  PatronDataRepository patronDataRepository,
                                  FinancialDataRepository financialDataRepository,
                                  DomainEventPublisher eventPublisher) {
        this.reportRepository = reportRepository;
        this.circulationDataRepository = circulationDataRepository;
        this.inventoryDataRepository = inventoryDataRepository;
        this.patronDataRepository = patronDataRepository;
        this.financialDataRepository = financialDataRepository;
        this.eventPublisher = eventPublisher;
    }
    
    /**
     * 生成借阅统计报表
     */
    @Transactional
    public AnalyticsReport generateCirculationReport(GenerateReportCommand command) {
        // 1. 创建报表记录
        ReportId reportId = ReportId.generate();
        AnalyticsReport report = new AnalyticsReport(
            reportId,
            ReportType.CIRCULATION_REPORT,
            command.getReportPeriod(),
            command.getReportDate(),
            command.getStartDate(),
            command.getEndDate(),
            command.getGeneratedBy()
        );
        
        reportRepository.save(report);
        
        // 2. 异步生成报表数据
        CompletableFuture.runAsync(() -> {
            try {
                generateCirculationReportData(report, command);
            } catch (Exception e) {
                report.fail(e.getMessage());
                reportRepository.save(report);
            }
        });
        
        return report;
    }

    // 私有方法：生成各类报表数据
    private void generateCirculationReportData(AnalyticsReport report, GenerateReportCommand command) {
        LocalDate startDate = command.getStartDate();
        LocalDate endDate = command.getEndDate();
        
        // 获取借阅数据
        CirculationStatistics stats = circulationDataRepository
            .getStatisticsByDateRange(startDate, endDate);
        
        // 添加指标
        report.addMetric("总借阅量", String.valueOf(stats.getTotalLoans()), "次");
        report.addMetric("总还书量", String.valueOf(stats.getTotalReturns()), "次");
        report.addMetric("平均借阅周期", stats.getAverageLoanPeriod() + " 天", "天");
        report.addMetric("逾期率", stats.getOverdueRate() + "%", "%");
        report.addMetric("续借率", stats.getRenewalRate() + "%", "%");
        
        // 添加维度
        report.addDimension("报表周期", command.getReportPeriod().name());
        report.addDimension("数据范围", startDate + " 至 " + endDate);
        
        // 生成趋势分析
        generateTrendAnalysis(report, "借阅量", stats.getTotalLoans(), 
                           circulationDataRepository.getPreviousPeriodLoans(startDate, endDate));
        
        // 完成报表
        String reportData = generateReportJson(stats);
        report.complete(stats.getTotalRecords(), "借阅统计报表生成成功", reportData);
        reportRepository.save(report);
    }
    
    private void generateInventoryReportData(AnalyticsReport report, GenerateReportCommand command) {
        LocalDate startDate = command.getStartDate();
        LocalDate endDate = command.getEndDate();
        
        InventoryStatistics stats = inventoryDataRepository
            .getStatisticsByDateRange(startDate, endDate);
        
        report.addMetric("总库存量", String.valueOf(stats.getTotalCopies()), "本");
        report.addMetric("可用副本", String.valueOf(stats.getAvailableCopies()), "本");
        report.addMetric("借出副本", String.valueOf(stats.getBorrowedCopies()), "本");
        report.addMetric("库存利用率", stats.getUtilizationRate() + "%", "%");
        report.addMetric("损坏丢失率", stats.getDamageLossRate() + "%", "%");
        
        report.addDimension("报表周期", command.getReportPeriod().name());
        report.addDimension("数据范围", startDate + " 至 " + endDate);
        
        String reportData = generateReportJson(stats);
        report.complete(stats.getTotalRecords(), "库存统计报表生成成功", reportData);
        reportRepository.save(report);
    }
    
    private void generatePopularityReportData(AnalyticsReport report, GenerateReportCommand command) {
        LocalDate startDate = command.getStartDate();
        LocalDate endDate = command.getEndDate();
        
        List<PopularBook> popularBooks = circulationDataRepository
            .getPopularBooks(startDate, endDate, 20);
        
        report.addMetric("热门图书数量", String.valueOf(popularBooks.size()), "本");
        
        for (int i = 0; i < popularBooks.size(); i++) {
            PopularBook book = popularBooks.get(i);
            report.addMetric("TOP" + (i + 1) + "借阅次数", 
                           String.valueOf(book.getBorrowCount()), "次");
            report.addDimension("TOP" + (i + 1) + "图书", book.getBookTitle());
        }
        
        String reportData = generateReportJson(popularBooks);
        report.complete((long) popularBooks.size(), "热门图书报表生成成功", reportData);
        reportRepository.save(report);
    }
    
    private void generateFinancialReportData(AnalyticsReport report, GenerateReportCommand command) {
        LocalDate startDate = command.getStartDate();
        LocalDate endDate = command.getEndDate();
        
        FinancialStatistics stats = financialDataRepository
            .getStatisticsByDateRange(startDate, endDate);
        
        report.addMetric("总收入", stats.getTotalIncome().toString(), "元");
        report.addMetric("罚金收入", stats.getFineIncome().toString(), "元");
        report.addMetric("会员费收入", stats.getMembershipIncome().toString(), "元");
        report.addMetric("服务费收入", stats.getServiceIncome().toString(), "元");
        report.addMetric("总支出", stats.getTotalExpense().toString(), "元");
        report.addMetric("净利润", stats.getNetProfit().toString(), "元");
        
        generateTrendAnalysis(report, "总收入", stats.getTotalIncome(), 
                           financialDataRepository.getPreviousPeriodIncome(startDate, endDate));
        
        String reportData = generateReportJson(stats);
        report.complete(stats.getTransactionCount(), "财务报表生成成功", reportData);
        reportRepository.save(report);
    }
    
    private void generateTrendAnalysis(AnalyticsReport report, String metricName, 
                                      BigDecimal currentValue, BigDecimal previousValue) {
        if (previousValue != null && previousValue.compareTo(BigDecimal.ZERO) > 0) {
            TrendAnalysis trend = new TrendAnalysis(metricName, currentValue, previousValue);
            // 将趋势分析保存到报表的额外数据中
            String trendInfo = String.format("{\"type\":\"%s\",\"direction\":\"%s\",\"change\":\"%s%%\",\"strength\":\"%s\"}",
                trend.getAnalysisType(),
                trend.getTrendDirection(),
                trend.getChangePercentage(),
                trend.getTrendStrength());
            
            report.addMetric(metricName + "趋势", trendInfo, "");
        }
    }
    
    private String generateReportJson(Object data) {
        // 使用JSON库生成报表数据
        // 简化示例，实际应使用Jackson或Gson
        return "{\"data\": " + data.toString() + "}";
    }
    
    /**
     * 生成实时仪表板数据
     */
    @Transactional(readOnly = true)
    public DashboardData generateDashboardData(DashboardId dashboardId) {
        // 获取仪表板配置
        Dashboard dashboard = dashboardRepository.findById(dashboardId)
            .orElseThrow(() -> new DashboardNotFoundException(dashboardId));
        
        // 生成各个小组件的数据
        DashboardData dashboardData = new DashboardData(dashboardId);
        
        for (DashboardWidget widget : dashboard.getWidgets()) {
            WidgetData widgetData = generateWidgetData(widget);
            dashboardData.addWidgetData(widgetData);
        }
        
        // 标记仪表板为已刷新
        dashboard.markAsRefreshed();
        dashboardRepository.save(dashboard);
        
        return dashboardData;
    }
    
    private WidgetData generateWidgetData(DashboardWidget widget) {
        switch (widget.getWidgetType()) {
            case "TOTAL_LOANS":
                return generateTotalLoansWidget(widget);
            case "ACTIVE_PATRONS":
                return generateActivePatronsWidget(widget);
            case "OVERDUE_BOOKS":
                return generateOverdueBooksWidget(widget);
            case "POPULAR_BOOKS":
                return generatePopularBooksWidget(widget);
            default:
                return generateDefaultWidget(widget);
        }
    }

package com.library.analytics.application.service;

import org.springframework.stereotype.Service;
import org.springframework.transaction.annotation.Transactional;

import java.util.List;
import java.util.stream.Collectors;

/**
 * 分析应用服务
 */
@Service
public class AnalyticsApplicationService {
    
    private final ReportGenerationService reportGenerationService;
    private final DashboardService dashboardService;
    private final AnalyticsReportRepository reportRepository;
    
    public AnalyticsApplicationService(ReportGenerationService reportGenerationService,
                                     DashboardService dashboardService,
                                     AnalyticsReportRepository reportRepository) {
        this.reportGenerationService = reportGenerationService;
        this.dashboardService = dashboardService;
        this.reportRepository = reportRepository;
    }
    
    /**
     * 生成报表用例
     */
    @Transactional
    public ReportDTO generateReport(GenerateReportRequest request) {
        GenerateReportCommand command = new GenerateReportCommand(
            request.getReportType(),
            request.getReportPeriod(),
            request.getReportDate(),
            request.getStartDate(),
            request.getEndDate(),
            request.getGeneratedBy()
        );
        
        AnalyticsReport report;
        switch (request.getReportType()) {
            case CIRCULATION_REPORT:
                report = reportGenerationService.generateCirculationReport(command);
                break;
            case INVENTORY_REPORT:
                report = reportGenerationService.generateInventoryReport(command);
                break;
            case POPULARITY_REPORT:
                report = reportGenerationService.generatePopularityReport(command);
                break;
            case FINANCIAL_REPORT:
                report = reportGenerationService.generateFinancialReport(command);
                break;
            default:
                throw new UnsupportedOperationException("Unsupported report type: " + request.getReportType());
        }
        
        return ReportDTO.fromDomain(report);
    }
    
    /**
     * 获取报表详情用例
     */
    public ReportDTO getReport(ReportId reportId) {
        AnalyticsReport report = reportRepository.findById(reportId)
            .orElseThrow(() -> new ReportNotFoundException(reportId));
        return ReportDTO.fromDomain(report);
    }
    
    /**
     * 获取仪表板数据用例
     */
    public DashboardDataDTO getDashboardData(DashboardId dashboardId) {
        DashboardData data = reportGenerationService.generateDashboardData(dashboardId);
        return DashboardDataDTO.fromDomain(data);
    }
    
    /**
     * 获取实时统计数据用例
     */
    public RealTimeStatisticsDTO getRealTimeStatistics() {
        return reportGenerationService.getRealTimeStatistics();
    }
    
    /**
     * 搜索报表用例
     */
    public Page<ReportDTO> searchReports(ReportSearchQuery query, Pageable pageable) {
        Page<AnalyticsReport> reports = reportRepository.search(query, pageable);
        return reports.map(ReportDTO::fromDomain);
    }

package com.library.analytics.infrastructure.datawarehouse;

/**
 * 数据仓库服务
 * 处理数据ETL和分析查询
 */
@Service
public class DataWarehouseService {
    
    private final JdbcTemplate jdbcTemplate;
    private final EntityManager entityManager;
    
    public DataWarehouseService(JdbcTemplate jdbcTemplate, EntityManager entityManager) {
        this.jdbcTemplate = jdbcTemplate;
        this.entityManager = entityManager;
    }
    
    /**
     * 执行ETL过程
     */
    @Scheduled(cron = "0 0 2 * * ?") // 每天凌晨2点执行
    public void executeETL() {
        LocalDateTime startTime = LocalDateTime.now();
        
        try {
            // 1. 提取数据
            extractData();
            
            // 2. 转换数据
            transformData();
            
            // 3. 加载数据
            loadData();
            
            // 4. 更新元数据
            updateMetadata(startTime);
            
        } catch (Exception e) {
            log.error("ETL process failed", e);
            // 发送告警
        }
    }
    
    private void extractData() {
        // 从各业务上下文提取数据
        extractCirculationData();
        extractInventoryData();
        extractPatronData();
        extractFinancialData();
    }
    
    private void transformData() {
        // 数据清洗和转换
        cleanAndTransformData();
        
        // 数据聚合
        aggregateData();
        
        // 计算派生指标
        calculateDerivedMetrics();
    }
    
    private void loadData() {
        // 加载到数据仓库表
        loadToFactTables();
        loadToDimensionTables();
    }
    
    /**
     * 执行分析查询
     */
    public List<Map<String, Object>> executeAnalysisQuery(String sql, Object... params) {
        return jdbcTemplate.queryForList(sql, params);
    }
    
    /**
     * 获取OLAP立方体数据
     */
    public OlapCubeResult getOlapCubeData(OlapCubeQuery query) {
        // 构建多维分析查询
        String mdxQuery = buildMdxQuery(query);
        
        // 执行查询
        List<Map<String, Object>> results = executeAnalysisQuery(mdxQuery);
        
        // 构建结果
        return new OlapCubeResult(query, results);
    }
}
