# 编目上下文（Catalog Context）详细设计

## 1. 上下文概述

### 1.1 职责范围
编目上下文负责图书馆的图书信息管理，包括：
- 图书基本信息的创建和维护
- ISBN验证和查询集成
- 图书分类体系管理
- 作者和出版社信息管理
- 图书元数据管理

### 1.2 业务价值
- 为其他上下文提供准确的图书信息
- 支持图书检索和分类浏览
- 维护图书知识的完整性
- 支持外部数据源集成

### 1.3 核心概念
- **Book（图书）**: 图书的基本信息单元
- **Author（作者）**: 图书的创作人员
- **Publisher（出版社）**: 图书的出版机构
- **Category（分类）**: 图书的分类标签
- **ISBN（国际标准书号）**: 图书的唯一标识符

```java
package com.library.catalog.domain.model;

import java.time.LocalDate;
import java.util.ArrayList;
import java.util.List;
import java.util.Objects;

/**
 * 图书聚合根
 * 负责管理图书的完整生命周期和业务不变量
 */
@Entity
@Table(name = "books")
public class Book {
    
    @EmbeddedId
    private BookId id;
    
    @Column(nullable = false, length = 500)
    private String title;
    
    @Column(nullable = false, length = 2000)
    private String description;
    
    @Embedded
    private ISBN isbn;
    
    @Column(nullable = false)
    private LocalDate publicationDate;
    
    @Column(nullable = false, length = 100)
    private String language;
    
    @Column(nullable = false)
    private Integer pageCount;
    
    @Column(name = "cover_image_url")
    private String coverImageUrl;
    
    @OneToMany(mappedBy = "book", cascade = CascadeType.ALL, orphanRemoval = true)
    private List<BookAuthor> authors = new ArrayList<>();
    
    @ManyToOne(fetch = FetchType.LAZY)
    @JoinColumn(name = "publisher_id")
    private Publisher publisher;
    
    @ManyToMany(fetch = FetchType.LAZY)
    @JoinTable(
        name = "book_categories",
        joinColumns = @JoinColumn(name = "book_id"),
        inverseJoinColumns = @JoinColumn(name = "category_id")
    )
    private Set<Category> categories = new HashSet<>();
    
    @Column(nullable = false)
    private BookStatus status;
    
    @Column(name = "created_at", nullable = false, updatable = false)
    private LocalDateTime createdAt;
    
    @Column(name = "updated_at")
    private LocalDateTime updatedAt;
    
    @Column(name = "created_by")
    private String createdBy;
    
    @Column(name = "updated_by")
    private String updatedBy;
    
    // 构造函数
    protected Book() {
        // JPA required
    }
    
    public Book(BookId id, String title, String description, ISBN isbn, 
                LocalDate publicationDate, String language, Integer pageCount,
                String createdBy) {
        validateBookData(title, description, isbn, publicationDate, language, pageCount);
        
        this.id = Objects.requireNonNull(id, "Book ID cannot be null");
        this.title = title;
        this.description = description;
        this.isbn = isbn;
        this.publicationDate = publicationDate;
        this.language = language;
        this.pageCount = pageCount;
        this.status = BookStatus.DRAFT;
        this.createdAt = LocalDateTime.now();
        this.updatedAt = LocalDateTime.now();
        this.createdBy = createdBy;
        this.updatedBy = createdBy;
    }
    
    // 领域行为：添加作者
    public void addAuthor(Author author, AuthorRole role) {
        Objects.requireNonNull(author, "Author cannot be null");
        Objects.requireNonNull(role, "Author role cannot be null");
        
        // 检查是否已存在该作者
        boolean authorExists = this.authors.stream()
            .anyMatch(ba -> ba.getAuthor().equals(author));
        
        if (authorExists) {
            throw new DuplicateAuthorException(this.id, author.getId());
        }
        
        BookAuthor bookAuthor = new BookAuthor(this, author, role);
        this.authors.add(bookAuthor);
        this.updatedAt = LocalDateTime.now();
    }
    
    // 领域行为：移除作者
    public void removeAuthor(AuthorId authorId) {
        Objects.requireNonNull(authorId, "Author ID cannot be null");
        
        boolean removed = this.authors.removeIf(ba -> 
            ba.getAuthor().getId().equals(authorId));
        
        if (!removed) {
            throw new AuthorNotFoundException(this.id, authorId);
        }
        
        if (this.authors.isEmpty()) {
            throw new InvalidOperationException("Book must have at least one author");
        }
        
        this.updatedAt = LocalDateTime.now();
    }
    
    // 领域行为：设置出版社
    public void setPublisher(Publisher publisher) {
        this.publisher = Objects.requireNonNull(publisher, "Publisher cannot be null");
        this.updatedAt = LocalDateTime.now();
    }
    
    // 领域行为：添加分类
    public void addCategory(Category category) {
        Objects.requireNonNull(category, "Category cannot be null");
        this.categories.add(category);
        this.updatedAt = LocalDateTime.now();
    }
    
    // 领域行为：移除分类
    public void removeCategory(CategoryId categoryId) {
        Objects.requireNonNull(categoryId, "Category ID cannot be null");
        boolean removed = this.categories.removeIf(c -> c.getId().equals(categoryId));
        this.updatedAt = LocalDateTime.now();
    }
    
    // 领域行为：更新基本信息
    public void updateBasicInfo(String title, String description, String updatedBy) {
        validateTitle(title);
        validateDescription(description);
        
        this.title = title;
        this.description = description;
        this.updatedBy = updatedBy;
        this.updatedAt = LocalDateTime.now();
    }
    
    // 领域行为：发布图书
    public void publish() {
        if (this.status != BookStatus.DRAFT) {
            throw new InvalidOperationException("Only draft books can be published");
        }
        
        if (this.authors.isEmpty()) {
            throw new InvalidOperationException("Book must have at least one author before publishing");
        }
        
        if (this.publisher == null) {
            throw new InvalidOperationException("Book must have a publisher before publishing");
        }
        
        if (this.categories.isEmpty()) {
            throw new InvalidOperationException("Book must have at least one category before publishing");
        }
        
        this.status = BookStatus.PUBLISHED;
        this.updatedAt = LocalDateTime.now();
    }
    
    // 领域行为：下架图书
    public void unpublish() {
        if (this.status != BookStatus.PUBLISHED) {
            throw new InvalidOperationException("Only published books can be unpublished");
        }
        
        this.status = BookStatus.UNPUBLISHED;
        this.updatedAt = LocalDateTime.now();
    }
    
    // 领域行为：删除图书
    public void delete() {
        if (this.status == BookStatus.PUBLISHED) {
            throw new InvalidOperationException("Published books cannot be deleted");
        }
        
        this.status = BookStatus.DELETED;
        this.updatedAt = LocalDateTime.now();
    }
    
    // 验证方法
    private void validateBookData(String title, String description, ISBN isbn, 
                                  LocalDate publicationDate, String language, Integer pageCount) {
        validateTitle(title);
        validateDescription(description);
        validateISBN(isbn);
        validatePublicationDate(publicationDate);
        validateLanguage(language);
        validatePageCount(pageCount);
    }
    
    private void validateTitle(String title) {
        if (title == null || title.trim().isEmpty()) {
            throw new IllegalArgumentException("Title cannot be empty");
        }
        if (title.length() > 500) {
            throw new IllegalArgumentException("Title cannot exceed 500 characters");
        }
    }
    
    private void validateDescription(String description) {
        if (description == null || description.trim().isEmpty()) {
            throw new IllegalArgumentException("Description cannot be empty");
        }
        if (description.length() > 2000) {
            throw new IllegalArgumentException("Description cannot exceed 2000 characters");
        }
    }
    
    private void validateISBN(ISBN isbn) {
        Objects.requireNonNull(isbn, "ISBN cannot be null");
        if (!isbn.isValid()) {
            throw new IllegalArgumentException("Invalid ISBN format");
        }
    }
    
    private void validatePublicationDate(LocalDate publicationDate) {
        Objects.requireNonNull(publicationDate, "Publication date cannot be null");
        if (publicationDate.isAfter(LocalDate.now())) {
            throw new IllegalArgumentException("Publication date cannot be in the future");
        }
    }
    
    private void validateLanguage(String language) {
        if (language == null || language.trim().isEmpty()) {
            throw new IllegalArgumentException("Language cannot be empty");
        }
    }
    
    private void validatePageCount(Integer pageCount) {
        Objects.requireNonNull(pageCount, "Page count cannot be null");
        if (pageCount <= 0) {
            throw new IllegalArgumentException("Page count must be positive");
        }
    }
    
    // Getters
    public BookId getId() { return id; }
    public String getTitle() { return title; }
    public String getDescription() { return description; }
    public ISBN getIsbn() { return isbn; }
    public LocalDate getPublicationDate() { return publicationDate; }
    public String getLanguage() { return language; }
    public Integer getPageCount() { return pageCount; }
    public String getCoverImageUrl() { return coverImageUrl; }
    public List<BookAuthor> getAuthors() { return Collections.unmodifiableList(authors); }
    public Publisher getPublisher() { return publisher; }
    public Set<Category> getCategories() { return Collections.unmodifiableSet(categories); }
    public BookStatus getStatus() { return status; }
    public LocalDateTime getCreatedAt() { return createdAt; }
    public LocalDateTime getUpdatedAt() { return updatedAt; }
    
    // 业务查询方法
    public boolean isPublished() {
        return this.status == BookStatus.PUBLISHED;
    }
    
    public boolean isDraft() {
        return this.status == BookStatus.DRAFT;
    }
    
    public String getMainAuthor() {
        return this.authors.stream()
            .filter(ba -> ba.getRole() == AuthorRole.PRIMARY_AUTHOR)
            .map(ba -> ba.getAuthor().getName())
            .findFirst()
            .orElse(this.authors.isEmpty() ? "" : this.authors.get(0).getAuthor().getName());
    }
}
```

### 2.2 值对象：ISBN

```java
package com.library.catalog.domain.model;

import java.util.Objects;
import java.util.regex.Pattern;

/**
 * ISBN值对象
 * 表示国际标准书号，提供验证和格式化功能
 */
@Embeddable
public class ISBN {
    
    private static final Pattern ISBN_10_PATTERN = Pattern.compile("^\\d{9}[\\dX]$");
    private static final Pattern ISBN_13_PATTERN = Pattern.compile("^\\d{13}$");
    
    @Column(name = "isbn", nullable = false, unique = true, length = 20)
    private String value;
    
    protected ISBN() {
        // JPA required
    }
    
    public ISBN(String value) {
        String cleanedValue = cleanISBN(value);
        validateISBN(cleanedValue);
        this.value = cleanedValue;
    }
    
    private String cleanISBN(String isbn) {
        if (isbn == null) {
            throw new IllegalArgumentException("ISBN cannot be null");
        }
        return isbn.replaceAll("[\\s-]", "").toUpperCase();
    }
    
    private void validateISBN(String isbn) {
        if (isbn.length() == 10) {
            if (!ISBN_10_PATTERN.matcher(isbn).matches()) {
                throw new IllegalArgumentException("Invalid ISBN-10 format");
            }
            if (!validateISBN10Checksum(isbn)) {
                throw new IllegalArgumentException("Invalid ISBN-10 checksum");
            }
        } else if (isbn.length() == 13) {
            if (!ISBN_13_PATTERN.matcher(isbn).matches()) {
                throw new IllegalArgumentException("Invalid ISBN-13 format");
            }
            if (!validateISBN13Checksum(isbn)) {
                throw new IllegalArgumentException("Invalid ISBN-13 checksum");
            }
        } else {
            throw new IllegalArgumentException("ISBN must be 10 or 13 digits");
        }
    }
    
    private boolean validateISBN10Checksum(String isbn) {
        int sum = 0;
        for (int i = 0; i < 9; i++) {
            sum += Character.getNumericValue(isbn.charAt(i)) * (10 - i);
        }
        char lastChar = isbn.charAt(9);
        int checksum = (lastChar == 'X') ? 10 : Character.getNumericValue(lastChar);
        return (sum + checksum) % 11 == 0;
    }
    
    private boolean validateISBN13Checksum(String isbn) {
        int sum = 0;
        for (int i = 0; i < 12; i++) {
            int digit = Character.getNumericValue(isbn.charAt(i));
            sum += (i % 2 == 0) ? digit : digit * 3;
        }
        int checksum = (10 - (sum % 10)) % 10;
        return checksum == Character.getNumericValue(isbn.charAt(12));
    }
    
    public boolean isValid() {
        try {
            validateISBN(this.value);
            return true;
        } catch (IllegalArgumentException e) {
            return false;
        }
    }
    
    public String getFormattedValue() {
        if (value.length() == 10) {
            return String.format("%s-%s-%s-%s", 
                value.substring(0, 1), value.substring(1, 4), 
                value.substring(4, 9), value.substring(9));
        } else {
            return String.format("%s-%s-%s-%s", 
                value.substring(0, 3), value.substring(3, 4), 
                value.substring(4, 9), value.substring(9));
        }
    }
    
    public String getValue() {
        return value;
    }
    
    public int getLength() {
        return value.length();
    }
    
    public boolean isISBN10() {
        return value.length() == 10;
    }
    
    public boolean isISBN13() {
        return value.length() == 13;
    }
    
    @Override
    public boolean equals(Object o) {
        if (this == o) return true;
        if (o == null || getClass() != o.getClass()) return false;
        ISBN isbn = (ISBN) o;
        return value.equals(isbn.value);
    }
    
    @Override
    public int hashCode() {
        return Objects.hash(value);
    }
    
    @Override
    public String toString() {
        return getFormattedValue();
    }
}
```

```java
package com.library.catalog.domain.model;

import javax.persistence.*;
import java.time.LocalDateTime;
import java.util.ArrayList;
import java.util.List;
import java.util.Objects;

/**
 * 分类实体
 * 表示图书的分类体系
 */
@Entity
@Table(name = "categories")
public class Category {
    
    @EmbeddedId
    private CategoryId id;
    
    @Column(nullable = false, length = 100)
    private String name;
    
    @Column(length = 500)
    private String description;
    
    @ManyToOne(fetch = FetchType.LAZY)
    @JoinColumn(name = "parent_id")
    private Category parent;
    
    @OneToMany(mappedBy = "parent", cascade = CascadeType.ALL)
    private List<Category> children = new ArrayList<>();
    
    @Column(nullable = false)
    private Integer level;
    
    @Column(length = 50)
    private String code;
    
    @ManyToMany(mappedBy = "categories")
    private List<Book> books = new ArrayList<>();
    
    @Column(name = "created_at", nullable = false, updatable = false)
    private LocalDateTime createdAt;
    
    @Column(name = "updated_at")
    private LocalDateTime updatedAt;
    
    protected Category() {
        // JPA required
    }
    
    public Category(CategoryId id, String name) {
        validateCategoryName(name);
        
        this.id = Objects.requireNonNull(id, "Category ID cannot be null");
        this.name = name;
        this.level = 0;
        this.createdAt = LocalDateTime.now();
        this.updatedAt = LocalDateTime.now();
    }
    
    public Category(CategoryId id, String name, Category parent) {
        validateCategoryName(name);
        Objects.requireNonNull(parent, "Parent category cannot be null for subcategory");
        
        this.id = Objects.requireNonNull(id, "Category ID cannot be null");
        this.name = name;
        this.parent = parent;
        this.level = parent.getLevel() + 1;
        this.createdAt = LocalDateTime.now();
        this.updatedAt = LocalDateTime.now();
    }
    
    public void addChild(Category child) {
        if (child.getLevel() != this.level + 1) {
            throw new IllegalArgumentException("Child category level must be parent level + 1");
        }
        this.children.add(child);
        this.updatedAt = LocalDateTime.now();
    }
    
    public void removeChild(CategoryId childId) {
        boolean removed = this.children.removeIf(c -> c.getId().equals(childId));
        this.updatedAt = LocalDateTime.now();
    }
    
    private void validateCategoryName(String name) {
        if (name == null || name.trim().isEmpty()) {
            throw new IllegalArgumentException("Category name cannot be empty");
        }
        if (name.length() > 100) {
            throw new IllegalArgumentException("Category name cannot exceed 100 characters");
        }
    }
    
    public boolean isRootCategory() {
        return this.parent == null;
    }
    
    public boolean isLeafCategory() {
        return this.children.isEmpty();
    }
    
    // Getters
    public CategoryId getId() { return id; }
    public String getName() { return name; }
    public String getDescription() { return description; }
    public Category getParent() { return parent; }
    public List<Category> getChildren() { return Collections.unmodifiableList(children); }
    public Integer getLevel() { return level; }
    public String getCode() { return code; }
}
```

```java
package com.library.catalog.domain.model;
```

```java
import org.springframework.stereotype.Service;
import org.springframework.transaction.annotation.Transactional;

import java.time.LocalDate;
import java.util.List;
import java.util.stream.Collectors;

/**
 * 图书管理领域服务
 * 处理跨聚合的复杂业务逻辑
 */
@Service
public class BookManagementService {
    
    private final BookRepository bookRepository;
    private final AuthorRepository authorRepository;
    private final PublisherRepository publisherRepository;
    private final CategoryRepository categoryRepository;
    private final DomainEventPublisher eventPublisher;
    private final ISBNValidationService isbnValidationService;
    
    public BookManagementService(BookRepository bookRepository,
                                  AuthorRepository authorRepository,
                                  PublisherRepository publisherRepository,
                                  CategoryRepository categoryRepository,
                                  DomainEventPublisher eventPublisher,
                                  ISBNValidationService isbnValidationService) {
        this.bookRepository = bookRepository;
        this.authorRepository = authorRepository;
        this.publisherRepository = publisherRepository;
        this.categoryRepository = categoryRepository;
        this.eventPublisher = eventPublisher;
        this.isbnValidationService = isbnValidationService;
    }
    
    /**
     * 创建新图书（草稿状态）
     */
    @Transactional
    public Book createBook(CreateBookCommand command) {
        // 1. 验证ISBN唯一性
        if (bookRepository.existsByISBN(command.getIsbn())) {
            throw new DuplicateISBNException(command.getIsbn());
        }
        
        // 2. 验证ISBN有效性
        if (!isbnValidationService.validate(command.getIsbn())) {
            throw new InvalidISBNException(command.getIsbn());
        }
        
        // 3. 创建图书聚合
        BookId bookId = BookId.generate();
        ISBN isbn = new ISBN(command.getIsbn());
        Book book = new Book(
            bookId,
            command.getTitle(),
            command.getDescription(),
            isbn,
            command.getPublicationDate(),
            command.getLanguage(),
            command.getPageCount(),
            command.getCreatedBy()
        );
        
        // 4. 添加作者
        for (AuthorInfo authorInfo : command.getAuthors()) {
            Author author = findOrCreateAuthor(authorInfo);
            book.addAuthor(author, authorInfo.getRole());
        }
        
        // 5. 设置出版社
        if (command.getPublisherId() != null) {
            Publisher publisher = publisherRepository.findById(command.getPublisherId())
                .orElseThrow(() -> new PublisherNotFoundException(command.getPublisherId()));
            book.setPublisher(publisher);
        }
        
        // 6. 添加分类
        for (CategoryId categoryId : command.getCategoryIds()) {
            Category category = categoryRepository.findById(categoryId)
                .orElseThrow(() -> new CategoryNotFoundException(categoryId));
            book.addCategory(category);
        }
        
        // 7. 保存图书
        Book savedBook = bookRepository.save(book);
        
        // 8. 发布领域事件
        eventPublisher.publish(new BookCreatedEvent(
            savedBook.getId(),
            savedBook.getIsbn(),
            savedBook.getTitle(),
            savedBook.getCreatedAt()
        ));
        
        return savedBook;
    }
    
    /**
     * 发布图书
     */
    @Transactional
    public void publishBook(BookId bookId, String publishedBy) {
        Book book = bookRepository.findById(bookId)
            .orElseThrow(() -> new BookNotFoundException(bookId));
        
        book.publish();
        bookRepository.save(book);
        
        eventPublisher.publish(new BookPublishedEvent(
            bookId,
            book.getTitle(),
            book.getIsbn(),
            LocalDateTime.now()
        ));
    }
    
    /**
     * 更新图书信息
     */
    @Transactional
    public void updateBook(UpdateBookCommand command) {
        Book book = bookRepository.findById(command.getBookId())
            .orElseThrow(() -> new BookNotFoundException(command.getBookId()));
        
        book.updateBasicInfo(
            command.getTitle(),
            command.getDescription(),
            command.getUpdatedBy()
        );
        
        bookRepository.save(book);
        
        eventPublisher.publish(new BookUpdatedEvent(
            command.getBookId(),
            LocalDateTime.now()
        ));
    }
    
    /**
     * 删除图书
     */
    @Transactional
    public void deleteBook(BookId bookId, String deletedBy) {
        Book book = bookRepository.findById(bookId)
            .orElseThrow(() -> new BookNotFoundException(bookId));
        
        book.delete();
        bookRepository.save(book);
        
        eventPublisher.publish(new BookDeletedEvent(
            bookId,
            book.getTitle(),
            LocalDateTime.now()
        ));
    }
    
    /**
     * 查找或创建作者
     */
    private Author findOrCreateAuthor(AuthorInfo authorInfo) {
        return authorRepository.findByName(authorInfo.getName())
            .orElseGet(() -> {
                AuthorId authorId = AuthorId.generate();
                Author author = new Author(authorId, authorInfo.getName());
                return authorRepository.save(author);
            });
    }
    
    /**
     * 批量导入图书
     */
    @Transactional
    public List<Book> importBooks(List<ImportBookCommand> commands, String importedBy) {
        return commands.stream()
            .map(command -> {
                CreateBookCommand createCommand = new CreateBookCommand(
                    command.getTitle(),
                    command.getDescription(),
                    command.getIsbn(),
                    command.getPublicationDate(),
                    command.getLanguage(),
                    command.getPageCount(),
                    command.getAuthors(),
                    command.getPublisherId(),
                    command.getCategoryIds(),
                    importedBy
                );
                return createBook(createCommand);
            })
            .collect(Collectors.toList());
    }
}
```

```java
package com.library.catalog.domain.service;

import com.library.catalog.domain.model.ISBN;
import org.springframework.stereotype.Service;

/**
 * ISBN验证领域服务
 * 提供ISBN验证和外部API查询功能
 */
@Service
public class ISBNValidationService {
    
    private final ExternalISBNService externalISBNService;
    
    public ISBNValidationService(ExternalISBNService externalISBNService) {
        this.externalISBNService = externalISBNService;
    }
    
    /**
     * 验证ISBN格式和校验和
     */
    public boolean validate(String isbnValue) {
        try {
            ISBN isbn = new ISBN(isbnValue);
            return isbn.isValid();
        } catch (IllegalArgumentException e) {
            return false;
        }
    }
    
    /**
     * 从外部服务获取ISBN信息
     */
    public ISBNInfo getISBNInfo(String isbnValue) {
        if (!validate(isbnValue)) {
            throw new InvalidISBNException(isbnValue);
        }
        
        return externalISBNService.fetchISBNInfo(isbnValue);
    }
    
    /**
     * 转换ISBN-10为ISBN-13
     */
    public String convertToISBN13(String isbn10) {
        if (!validate(isbn10)) {
            throw new InvalidISBNException(isbn10);
        }
        
        ISBN isbn = new ISBN(isbn10);
        if (!isbn.isISBN10()) {
            throw new IllegalArgumentException("ISBN must be ISBN-10 format");
        }
        
        // ISBN-13转换逻辑：添加978前缀，重新计算校验位
        String isbn13WithoutCheck = "978" + isbn10.substring(0, 9);
        int checksum = calculateISBN13Checksum(isbn13WithoutCheck);
        return isbn13WithoutCheck + checksum;
    }
    
    private int calculateISBN13Checksum(String isbn12) {
        int sum = 0;
        for (int i = 0; i < 12; i++) {
            int digit = Character.getNumericValue(isbn12.charAt(i));
            sum += (i % 2 == 0) ? digit : digit * 3;
        }
        return (10 - (sum % 10)) % 10;
    }
}
```

```java
package com.library.catalog.interfaces.rest;

import com.library.catalog.application.dto.*;
import com.library.catalog.application.service.BookApplicationService;
import com.library.catalog.domain.model.BookId;
import org.springframework.data.domain.Page;
import org.springframework.data.domain.PageRequest;
import org.springframework.data.domain.Pageable;
import org.springframework.data.domain.Sort;
import org.springframework.http.HttpStatus;
import org.springframework.http.ResponseEntity;
import org.springframework.web.bind.annotation.*;

import javax.validation.Valid;

/**
 * 图书REST控制器
 * 提供图书管理的HTTP API
 */
@RestController
@RequestMapping("/api/catalog/books")
public class BookController {
    
    private final BookApplicationService bookApplicationService;
    
    public BookController(BookApplicationService bookApplicationService) {
        this.bookApplicationService = bookApplicationService;
    }
    
    /**
     * 创建图书
     * POST /api/catalog/books
     */
    @PostMapping
    public ResponseEntity<ApiResponse<BookDTO>> createBook(
            @Valid @RequestBody CreateBookRequest request) {
        BookDTO book = bookApplicationService.createBook(request);
        return ResponseEntity.status(HttpStatus.CREATED)
            .body(ApiResponse.success(book));
    }
    
    /**
     * 获取图书详情
     * GET /api/catalog/books/{id}
     */
    @GetMapping("/{id}")
    public ResponseEntity<ApiResponse<BookDTO>> getBook(@PathVariable String id) {
        BookId bookId = BookId.of(id);
        BookDTO book = bookApplicationService.getBook(bookId);
        return ResponseEntity.ok(ApiResponse.success(book));
    }
    
    /**
     * 搜索图书
     * GET /api/catalog/books?title=xxx&author=xxx&page=0&size=10
     */
    @GetMapping
    public ResponseEntity<ApiResponse<Page<BookDTO>>> searchBooks(
            @RequestParam(required = false) String title,
            @RequestParam(required = false) String author,
            @RequestParam(required = false) String isbn,
            @RequestParam(defaultValue = "0") int page,
            @RequestParam(defaultValue = "10") int size,
            @RequestParam(defaultValue = "title") String sortBy,
            @RequestParam(defaultValue = "asc") String sortDirection) {
        
        BookSearchQuery query = new BookSearchQuery(title, author, isbn);
        
        Sort sort = Sort.by(Sort.Direction.fromString(sortDirection), sortBy);
        Pageable pageable = PageRequest.of(page, size, sort);
        
        Page<BookDTO> books = bookApplicationService.searchBooks(query, pageable);
        return ResponseEntity.ok(ApiResponse.success(books));
    }
    
    /**
     * 发布图书
     * POST /api/catalog/books/{id}/publish
     */
    @PostMapping("/{id}/publish")
    public ResponseEntity<ApiResponse<Void>> publishBook(
            @PathVariable String id,
            @Valid @RequestBody PublishBookRequest request) {
        BookId bookId = BookId.of(id);
        request.setBookId(bookId);
        bookApplicationService.publishBook(request);
        return ResponseEntity.ok(ApiResponse.success(null));
    }
}
```

```java
package com.library.catalog.interfaces.rest;

import com.library.catalog.domain.exception.*;
import org.springframework.http.HttpStatus;
import org.springframework.http.ResponseEntity;
import org.springframework.web.bind.annotation.ExceptionHandler;
import org.springframework.web.bind.annotation.RestControllerAdvice;

@RestControllerAdvice
public class CatalogExceptionHandler {
    
    @ExceptionHandler(BookNotFoundException.class)
    public ResponseEntity<ApiResponse<Void>> handleBookNotFound(BookNotFoundException ex) {
        return ResponseEntity.status(HttpStatus.NOT_FOUND)
            .body(ApiResponse.error("Book not found: " + ex.getMessage()));
    }
    
    @ExceptionHandler(DuplicateISBNException.class)
    public ResponseEntity<ApiResponse<Void>> handleDuplicateISBN(DuplicateISBNException ex) {
        return ResponseEntity.status(HttpStatus.CONFLICT)
            .body(ApiResponse.error("Duplicate ISBN: " + ex.getMessage()));
    }
    
    @ExceptionHandler(InvalidISBNException.class)
    public ResponseEntity<ApiResponse<Void>> handleInvalidISBN(InvalidISBNException ex) {
        return ResponseEntity.status(HttpStatus.BAD_REQUEST)
            .body(ApiResponse.error("Invalid ISBN: " + ex.getMessage()));
    }
}
```
