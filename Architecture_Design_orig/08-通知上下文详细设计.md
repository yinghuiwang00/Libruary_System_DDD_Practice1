# 通知上下文（Notification Context）详细设计

## 1. 上下文概述

### 1.1 职责范围
通知上下文负责图书馆系统的所有消息通知功能，包括：
- 借阅到期提醒
- 逾期通知
- 预约通知
- 罚金通知
- 系统公告
- 多渠道通知（邮件/短信/推送）
- 通知模板管理
- 通知历史记录
- 通知偏好设置

### 1.2 业务价值
- 及时提醒读者重要信息
- 提高读者满意度和体验
- 减少逾期和罚金
- 支持个性化通知
- 确保信息有效传达

### 1.3 核心概念
- **Notification（通知）**: 单次通知记录
- **NotificationChannel（通知渠道）**: 通知发送方式
- **NotificationTemplate（通知模板）**: 消息模板
- **NotificationPreference（通知偏好）**: 读者通知设置
- **DeliveryStatus（投递状态）**: 通知发送状态
- **NotificationQueue（通知队列）**: 异步通知队列

### 2.1 聚合根：Notification

```java
package com.library.notification.domain.model;

import javax.persistence.*;
import java.time.LocalDateTime;
import java.util.ArrayList;
import java.util.List;
import java.util.Objects;

/**
 * 通知聚合根
 * 管理通知的完整生命周期
 */
@Entity
@Table(name = "notifications", indexes = {
    @Index(name = "idx_notification_recipient", columnList = "recipient_id"),
    @Index(name = "idx_notification_type", columnList = "notification_type"),
    @Index(name = "idx_notification_status", columnList = "status"),
    @Index(name = "idx_notification_created", columnList = "created_at")
})
public class Notification {
    
    @EmbeddedId
    private NotificationId id;
    
    @Enumerated(EnumType.STRING)
    @Column(nullable = false)
    private NotificationType notificationType;
    
    @Enumerated(EnumType.STRING)
    @Column(nullable = false)
    private NotificationPriority priority;
    
    @Embedded
    @AttributeOverride(name = "value", column = @Column(name = "recipient_id"))
    private RecipientId recipientId;
    
    @Column(name = "recipient_email", length = 200)
    private String recipientEmail;
    
    @Column(name = "recipient_phone", length = 20)
    private String recipientPhone;
    
    @Column(name = "subject", length = 500)
    private String subject;
    
    @Column(name = "content", columnDefinition = "TEXT")
    private String content;
    
    @Enumerated(EnumType.STRING)
    @Column(nullable = false)
    private NotificationStatus status;
    
    @Column(name = "scheduled_at")
    private LocalDateTime scheduledAt;
    
    @Column(name = "sent_at")
    private LocalDateTime sentAt;
    
    @Column(name = "delivered_at")
    private LocalDateTime deliveredAt;
    
    @Column(name = "read_at")
    private LocalDateTime readAt;
    
    @Column(name = "failed_at")
    private LocalDateTime failedAt;
    
    @Column(name = "failure_reason", length = 500)
    private String failureReason;
    
    @Column(name = "retry_count")
    private Integer retryCount;
    
    @Column(name = "max_retries")
    private Integer maxRetries;
    
    @OneToMany(mappedBy = "notification", cascade = CascadeType.ALL)
    private List<NotificationDelivery> deliveries = new ArrayList<>();
    
    @Column(name = "template_id", length = 100)
    private String templateId;
    
    @Column(name = "template_variables", columnDefinition = "TEXT")
    private String templateVariables;
    
    @Column(name = "created_at", nullable = false, updatable = false)
    private LocalDateTime createdAt;
    
    @Column(name = "updated_at")
    private LocalDateTime updatedAt;
    
    protected Notification() {
        // JPA required
    }
    
    public Notification(NotificationId id, NotificationType notificationType, 
                      RecipientId recipientId, String subject, String content) {
        this.id = Objects.requireNonNull(id, "Notification ID cannot be null");
        this.notificationType = Objects.requireNonNull(notificationType, "Notification type cannot be null");
        this.priority = NotificationPriority.NORMAL;
        this.recipientId = Objects.requireNonNull(recipientId, "Recipient ID cannot be null");
        this.subject = subject;
        this.content = Objects.requireNonNull(content, "Content cannot be null");
        this.status = NotificationStatus.PENDING;
        this.retryCount = 0;
        this.maxRetries = 3;
        this.createdAt = LocalDateTime.now();
        this.updatedAt = LocalDateTime.now();
    }
    
    // 领域行为：调度通知
    public void schedule(LocalDateTime scheduledAt) {
        if (this.status != NotificationStatus.PENDING) {
            throw new InvalidOperationException("Can only schedule pending notifications");
        }
        
        if (scheduledAt.isBefore(LocalDateTime.now())) {
            throw new IllegalArgumentException("Scheduled time must be in the future");
        }
        
        this.scheduledAt = scheduledAt;
        this.status = NotificationStatus.SCHEDULED;
        this.updatedAt = LocalDateTime.now();
    }
    
    // 领域行为：发送通知
    public void send() {
        if (this.status != NotificationStatus.PENDING && this.status != NotificationStatus.SCHEDULED) {
            throw new InvalidOperationException("Notification is not ready to send");
        }
        
        this.status = NotificationStatus.SENDING;
        this.sentAt = LocalDateTime.now();
        this.updatedAt = LocalDateTime.now();
        
        DomainEventPublisher.publish(new NotificationSendingEvent(
            this.id,
            this.notificationType,
            this.recipientId,
            LocalDateTime.now()
        ));
    }
    
    // 领域行为：标记为已投递
    public void markAsDelivered(String channel) {
        if (this.status != NotificationStatus.SENDING) {
            throw new InvalidOperationException("Notification is not in sending status");
        }
        
        this.status = NotificationStatus.DELIVERED;
        this.deliveredAt = LocalDateTime.now();
        this.updatedAt = LocalDateTime.now();
        
        // 记录投递信息
        NotificationDelivery delivery = new NotificationDelivery(
            this,
            channel,
            DeliveryStatus.DELIVERED,
            LocalDateTime.now()
        );
        this.deliveries.add(delivery);
        
        DomainEventPublisher.publish(new NotificationDeliveredEvent(
            this.id,
            this.notificationType,
            this.recipientId,
            channel,
            this.deliveredAt
        ));
    }
    
    // 领域行为：标记为已读
    public void markAsRead() {
        if (this.status != NotificationStatus.DELIVERED) {
            throw new InvalidOperationException("Can only mark delivered notifications as read");
        }
        
        this.status = NotificationStatus.READ;
        this.readAt = LocalDateTime.now();
        this.updatedAt = LocalDateTime.now();
        
        DomainEventPublisher.publish(new NotificationReadEvent(
            this.id,
            this.notificationType,
            this.recipientId,
            this.readAt
        ));
    }
    
    // 领域行为：发送失败
    public void markAsFailed(String failureReason) {
        if (this.status != NotificationStatus.SENDING && this.status != NotificationStatus.SCHEDULED) {
            throw new InvalidOperationException("Notification is not in sendable status");
        }
        
        this.status = NotificationStatus.FAILED;
        this.failedAt = LocalDateTime.now();
        this.failureReason = failureReason;
        this.retryCount++;
        this.updatedAt = LocalDateTime.now();
        
        // 记录失败信息
        NotificationDelivery delivery = new NotificationDelivery(
            this,
            "EMAIL", // 默认渠道
            DeliveryStatus.FAILED,
            LocalDateTime.now()
        );
        delivery.setErrorMessage(failureReason);
        this.deliveries.add(delivery);
        
        DomainEventPublisher.publish(new NotificationFailedEvent(
            this.id,
            this.notificationType,
            this.recipientId,
            failureReason,
            this.retryCount,
            LocalDateTime.now()
        ));
    }
    
    // 领域行为：重试发送
    public void retry() {
        if (this.status != NotificationStatus.FAILED) {
            throw new InvalidOperationException("Can only retry failed notifications");
        }
        
        if (this.retryCount >= this.maxRetries) {
            throw new MaxRetriesExceededException("Maximum retry attempts exceeded");
        }
        
        this.status = NotificationStatus.PENDING;
        this.failureReason = null;
        this.updatedAt = LocalDateTime.now();
        
        DomainEventPublisher.publish(new NotificationRetryEvent(
            this.id,
            this.notificationType,
            this.recipientId,
            this.retryCount,
            LocalDateTime.now()
        ));
    }
    
    // 领域行为：取消通知
    public void cancel(String reason) {
        if (this.status == NotificationStatus.DELIVERED || this.status == NotificationStatus.READ) {
            throw new InvalidOperationException("Cannot cancel delivered or read notifications");
        }
        
        NotificationStatus oldStatus = this.status;
        this.status = NotificationStatus.CANCELLED;
        this.failureReason = reason;
        this.updatedAt = LocalDateTime.now();
        
        DomainEventPublisher.publish(new NotificationCancelledEvent(
            this.id,
            this.notificationType,
            this.recipientId,
            oldStatus,
            reason,
            LocalDateTime.now()
        ));
    }
    
    // 领域行为：设置优先级
    public void setPriority(NotificationPriority priority) {
        if (this.status != NotificationStatus.PENDING && this.status != NotificationStatus.SCHEDULED) {
            throw new InvalidOperationException("Can only set priority for pending or scheduled notifications");
        }
        
        this.priority = priority;
        this.updatedAt = LocalDateTime.now();
    }
    
    // 领域行为：设置模板信息
    public void setTemplate(String templateId, Map<String, Object> variables) {
        if (this.status != NotificationStatus.PENDING) {
            throw new InvalidOperationException("Can only set template for pending notifications");
        }
        
        this.templateId = templateId;
        this.templateVariables = serializeTemplateVariables(variables);
        this.updatedAt = LocalDateTime.now();
    }
    
    // 业务查询方法
    public boolean isPending() {
        return this.status == NotificationStatus.PENDING || this.status == NotificationStatus.SCHEDULED;
    }
    
    public boolean isDelivered() {
        return this.status == NotificationStatus.DELIVERED || this.status == NotificationStatus.READ;
    }
    
    public boolean isFailed() {
        return this.status == NotificationStatus.FAILED;
    }
    
    public boolean canBeRetried() {
        return this.status == NotificationStatus.FAILED && this.retryCount < this.maxRetries;
    }
    
    public boolean isOverdue() {
        if (this.scheduledAt == null) {
            return false;
        }
        return this.scheduledAt.isBefore(LocalDateTime.now()) && this.status == NotificationStatus.SCHEDULED;
    }
    
    public long getAgeMinutes() {
        return java.time.Duration.between(this.createdAt, LocalDateTime.now()).toMinutes();
    }
    
    // Getters
    public NotificationId getId() { return id; }
    public NotificationType getNotificationType() { return notificationType; }
    public NotificationPriority getPriority() { return priority; }
    public RecipientId getRecipientId() { return recipientId; }
    public String getRecipientEmail() { return recipientEmail; }
    public String getRecipientPhone() { return recipientPhone; }
    public String getSubject() { return subject; }
    public String getContent() { return content; }
    public NotificationStatus getStatus() { return status; }
    public LocalDateTime getScheduledAt() { return scheduledAt; }
    public LocalDateTime getSentAt() { return sentAt; }
    public LocalDateTime getDeliveredAt() { return deliveredAt; }
    public LocalDateTime getReadAt() { return readAt; }
    public String getFailureReason() { return failureReason; }
    public Integer getRetryCount() { return retryCount; }
    public List<NotificationDelivery> getDeliveries() { return new ArrayList<>(deliveries); }
    public String getTemplateId() { return templateId; }
}
```

```java
package com.library.notification.domain.model;

import javax.persistence.*;
import java.time.LocalDateTime;
import java.util.Objects;

/**
 * 通知模板实体
 * 定义可重用的消息模板
 */
@Entity
@Table(name = "notification_templates")
public class NotificationTemplate {
    
    @EmbeddedId
    private TemplateId id;
    
    @Column(nullable = false, length = 100, unique = true)
    private String templateCode;
    
    @Column(nullable = false, length = 200)
    private String templateName;
    
    @Enumerated(EnumType.STRING)
    @Column(nullable = false)
    private NotificationType notificationType;
    
    @Enumerated(EnumType.STRING)
    @Column(nullable = false)
    private NotificationChannel defaultChannel;
    
    @Column(name = "subject_template", length = 500)
    private String subjectTemplate;
    
    @Column(name = "content_template", columnDefinition = "TEXT")
    private String contentTemplate;
    
    @Column(name = "is_active")
    private Boolean active;
    
    @Column(name = "supported_variables", columnDefinition = "TEXT")
    private String supportedVariables;
    
    @Column(name = "created_at", nullable = false, updatable = false)
    private LocalDateTime createdAt;
    
    @Column(name = "updated_at")
    private LocalDateTime updatedAt;
    
    @Column(name = "created_by")
    private String createdBy;
    
    protected NotificationTemplate() {
        // JPA required
    }
    
    public NotificationTemplate(TemplateId id, String templateCode, String templateName,
                               NotificationType notificationType, NotificationChannel defaultChannel,
                               String subjectTemplate, String contentTemplate, String createdBy) {
        this.id = Objects.requireNonNull(id, "Template ID cannot be null");
        this.templateCode = validateTemplateCode(templateCode);
        this.templateName = Objects.requireNonNull(templateName, "Template name cannot be null");
        this.notificationType = Objects.requireNonNull(notificationType, "Notification type cannot be null");
        this.defaultChannel = Objects.requireNonNull(defaultChannel, "Default channel cannot be null");
        this.subjectTemplate = subjectTemplate;
        this.contentTemplate = Objects.requireNonNull(contentTemplate, "Content template cannot be null");
        this.active = true;
        this.createdAt = LocalDateTime.now();
        this.updatedAt = LocalDateTime.now();
        this.createdBy = createdBy;
    }
    
    // 领域行为：渲染通知
    public Notification render(RecipientId recipientId, Map<String, Object> variables) {
        String subject = renderTemplate(this.subjectTemplate, variables);
        String content = renderTemplate(this.contentTemplate, variables);
        
        NotificationId notificationId = NotificationId.generate();
        Notification notification = new Notification(
            notificationId,
            this.notificationType,
            recipientId,
            subject,
            content
        );
        
        notification.setTemplate(this.templateCode, variables);
        
        return notification;
    }
    
    // 领域行为：更新模板
    public void updateTemplate(String subjectTemplate, String contentTemplate, String updatedBy) {
        this.subjectTemplate = subjectTemplate;
        this.contentTemplate = Objects.requireNonNull(contentTemplate, "Content template cannot be null");
        this.updatedBy = updatedBy;
        this.updatedAt = LocalDateTime.now();
    }
    
    // 领域行为：激活/停用模板
    public void setActive(boolean active) {
        this.active = active;
        this.updatedAt = LocalDateTime.now();
    }
    
    // 领域行为：添加支持的变量
    public void addSupportedVariable(String variableName, String description, String dataType) {
        // 实现变量定义
        this.updatedAt = LocalDateTime.now();
    }
    
    // 私有方法
    private String renderTemplate(String template, Map<String, Object> variables) {
        String result = template;
        
        if (variables != null) {
            for (Map.Entry<String, Object> entry : variables.entrySet()) {
                String placeholder = "${" + entry.getKey() + "}";
                String value = entry.getValue() != null ? entry.getValue().toString() : "";
                result = result.replace(placeholder, value);
            }
        }
        
        return result;
    }
    
    private String validateTemplateCode(String code) {
        if (code == null || code.trim().isEmpty()) {
            throw new IllegalArgumentException("Template code cannot be empty");
        }
        if (!code.matches("^[A-Z_][A-Z0-9_]*$")) {
            throw new IllegalArgumentException("Template code must be uppercase with underscores");
        }
        return code.toUpperCase();
    }
    
    // 业务查询方法
    public boolean isActive() {
        return this.active != null && this.active;
    }
    
    // Getters
    public TemplateId getId() { return id; }
    public String getTemplateCode() { return templateCode; }
    public String getTemplateName() { return templateName; }
    public NotificationType getNotificationType() { return notificationType; }
    public NotificationChannel getDefaultChannel() { return defaultChannel; }
    public String getSubjectTemplate() { return subjectTemplate; }
    public String getContentTemplate() { return contentTemplate; }
    public Boolean getActive() { return active; }
}
```

```java
package com.library.notification.domain.model;

import javax.persistence.*;
import java.time.LocalDateTime;
import java.util.ArrayList;
import java.util.List;
import java.util.Objects;

/**
 * 通知偏好实体
 * 管理读者的通知设置
 */
@Entity
@Table(name = "notification_preferences")
public class NotificationPreference {
    
    @EmbeddedId
    private PreferenceId id;
    
    @Embedded
    @AttributeOverride(name = "value", column = @Column(name = "patron_id"))
    private PatronId patronId;
    
    @Column(name = "email_enabled")
    private Boolean emailEnabled;
    
    @Column(name = "sms_enabled")
    private Boolean smsEnabled;
    
    @Column(name = "push_enabled")
    private Boolean pushEnabled;
    
    @Column(name = "email_address", length = 200)
    private String emailAddress;
    
    @Column(name = "phone_number", length = 20)
    private String phoneNumber;
    
    @Column(name = "device_token", length = 500)
    private String deviceToken;
    
    @Enumerated(EnumType.STRING)
    @Column(name = "notification_timing")
    private NotificationTiming notificationTiming;
    
    @Column(name = "quiet_hours_start", length = 10)
    private String quietHoursStart;
    
    @Column(name = "quiet_hours_end", length = 10)
    private String quietHoursEnd;
    
    @OneToMany(mappedBy = "preference", cascade = CascadeType.ALL)
    private List<PreferenceTypeSetting> typeSettings = new ArrayList<>();
    
    @Column(name = "created_at", nullable = false, updatable = false)
    private LocalDateTime createdAt;
    
    @Column(name = "updated_at")
    private LocalDateTime updatedAt;
    
    protected NotificationPreference() {
        // JPA required
    }
    
    public NotificationPreference(PreferenceId id, PatronId patronId) {
        this.id = Objects.requireNonNull(id, "Preference ID cannot be null");
        this.patronId = Objects.requireNonNull(patronId, "Patron ID cannot be null");
        this.emailEnabled = true;
        this.smsEnabled = false;
        this.pushEnabled = true;
        this.notificationTiming = NotificationTiming.IMMEDIATE;
        this.quietHoursStart = "22:00";
        this.quietHoursEnd = "08:00";
        this.createdAt = LocalDateTime.now();
        this.updatedAt = LocalDateTime.now();
        
        // 添加默认类型设置
        addDefaultTypeSettings();
    }
    
    // 领域行为：启用邮件通知
    public void enableEmail(String emailAddress) {
        this.emailEnabled = true;
        this.emailAddress = emailAddress;
        this.updatedAt = LocalDateTime.now();
    }
    
    // 领域行为：禁用邮件通知
    public void disableEmail() {
        this.emailEnabled = false;
        this.updatedAt = LocalDateTime.now();
    }
    
    // 领域行为：启用短信通知
    public void enableSms(String phoneNumber) {
        this.smsEnabled = true;
        this.phoneNumber = phoneNumber;
        this.updatedAt = LocalDateTime.now();
    }
    
    // 领域行为：禁用短信通知
    public void disableSms() {
        this.smsEnabled = false;
        this.updatedAt = LocalDateTime.now();
    }
    
    // 领域行为：设置通知时机
    public void setNotificationTiming(NotificationTiming timing) {
        this.notificationTiming = Objects.requireNonNull(timing, "Notification timing cannot be null");
        this.updatedAt = LocalDateTime.now();
    }
    
    // 领域行为：设置免打扰时间
    public void setQuietHours(String startTime, String endTime) {
        validateTimeFormat(startTime);
        validateTimeFormat(endTime);
        this.quietHoursStart = startTime;
        this.quietHoursEnd = endTime;
        this.updatedAt = LocalDateTime.now();
    }
    
    // 领域行为：添加类型设置
    public void addTypeSetting(NotificationType notificationType, boolean enabled, 
                             NotificationChannel preferredChannel) {
        PreferenceTypeSetting setting = new PreferenceTypeSetting(
            this,
            notificationType,
            enabled,
            preferredChannel
        );
        this.typeSettings.add(setting);
        this.updatedAt = LocalDateTime.now();
    }
    
    // 领域行为：更新类型设置
    public void updateTypeSetting(NotificationType notificationType, boolean enabled, 
                                NotificationChannel preferredChannel) {
        PreferenceTypeSetting setting = findTypeSetting(notificationType);
        if (setting != null) {
            setting.setEnabled(enabled);
            setting.setPreferredChannel(preferredChannel);
            this.updatedAt = LocalDateTime.now();
        } else {
            addTypeSetting(notificationType, enabled, preferredChannel);
        }
    }
    
    // 业务查询方法
    public boolean isChannelEnabled(NotificationChannel channel) {
        switch (channel) {
            case EMAIL:
                return this.emailEnabled != null && this.emailEnabled;
            case SMS:
                return this.smsEnabled != null && this.smsEnabled;
            case PUSH:
                return this.pushEnabled != null && this.pushEnabled;
            default:
                return false;
        }
    }
    
    public boolean isTypeEnabled(NotificationType notificationType) {
        PreferenceTypeSetting setting = findTypeSetting(notificationType);
        return setting != null && setting.getEnabled();
    }
    
    public boolean isInQuietHours() {
        if (this.notificationTiming != NotificationTiming.IMMEDIATE) {
            return false;
        }
        
        LocalTime now = LocalTime.now();
        LocalTime quietStart = LocalTime.parse(this.quietHoursStart);
        LocalTime quietEnd = LocalTime.parse(this.quietHoursEnd);
        
        if (quietStart.isBefore(quietEnd)) {
            return now.isAfter(quietStart) && now.isBefore(quietEnd);
        } else {
            // 跨越午夜的情况（如22:00-08:00）
            return now.isAfter(quietStart) || now.isBefore(quietEnd);
        }
    }
    
    public NotificationChannel getPreferredChannel(NotificationType notificationType) {
        PreferenceTypeSetting setting = findTypeSetting(notificationType);
        if (setting != null && setting.getPreferredChannel() != null) {
            return setting.getPreferredChannel();
        }
        
        // 返回默认偏好
        if (this.emailEnabled) {
            return NotificationChannel.EMAIL;
        } else if (this.pushEnabled) {
            return NotificationChannel.PUSH;
        } else if (this.smsEnabled) {
            return NotificationChannel.SMS;
        }
        
        return NotificationChannel.EMAIL;
    }
    
    // 私有方法
    private void addDefaultTypeSettings() {
        addTypeSetting(NotificationType.DUE_DATE_REMINDER, true, NotificationChannel.EMAIL);
        addTypeSetting(NotificationType.OVERDUE_NOTICE, true, NotificationChannel.EMAIL);
        addTypeSetting(NotificationType.HOLD_AVAILABLE, true, NotificationChannel.EMAIL);
        addTypeSetting(NotificationType.FINE_NOTIFICATION, true, NotificationChannel.EMAIL);
    }
    
    private PreferenceTypeSetting findTypeSetting(NotificationType notificationType) {
        return this.typeSettings.stream()
            .filter(s -> s.getNotificationType() == notificationType)
            .findFirst()
            .orElse(null);
    }
    
    private void validateTimeFormat(String time) {
        try {
            LocalTime.parse(time);
        } catch (Exception e) {
            throw new IllegalArgumentException("Invalid time format. Expected HH:mm");
        }
    }
    
    // Getters
    public PreferenceId getId() { return id; }
    public PatronId getPatronId() { return patronId; }
    public Boolean getEmailEnabled() { return emailEnabled; }
    public Boolean getSmsEnabled() { return smsEnabled; }
    public Boolean getPushEnabled() { return pushEnabled; }
    public String getEmailAddress() { return emailAddress; }
    public String getPhoneNumber() { return phoneNumber; }
    public String getDeviceToken() { return deviceToken; }
    public NotificationTiming getNotificationTiming() { return notificationTiming; }
    public String getQuietHoursStart() { return quietHoursStart; }
    public String getQuietHoursEnd() { return quietHoursEnd; }
    public List<PreferenceTypeSetting> getTypeSettings() { return new ArrayList<>(typeSettings); }
}
```

```java
package com.library.notification.domain.model;

/**
 * 通知类型枚举
 */
public enum NotificationType {
    DUE_DATE_REMINDER,      // 到期提醒
    OVERDUE_NOTICE,         // 逾期通知
    HOLD_AVAILABLE,         // 预约可用
    HOLD_CANCELLED,         // 预约取消
    FINE_NOTIFICATION,      // 罚金通知
    PAYMENT_CONFIRMATION,   // 支付确认
    BOOK_RETURNED,          // 图书归还
    MEMBERSHIP_RENEWAL,     // 会员续费
    SYSTEM_ANNOUNCEMENT,    // 系统公告
    MARKETING_PROMOTION     // 营销推广
}

/**
 * 通知渠道枚举
 */
public enum NotificationChannel {
```

## 3. 领域服务设计

### 3.1 通知发送领域服务

```java
package com.library.notification.domain.service;

import com.library.notification.domain.model.*;
import com.library.notification.domain.repository.*;
import com.library.notification.domain.event.*;
import com.library.shared.domain.event.DomainEventPublisher;
import org.springframework.stereotype.Service;
import org.springframework.transaction.annotation.Transactional;

import java.time.LocalDateTime;
import java.util.List;
import java.util.Map;

/**
 * 通知发送领域服务
 */
@Service
public class NotificationSendingService {
    
    private final NotificationRepository notificationRepository;
    private final NotificationTemplateRepository templateRepository;
    private final NotificationPreferenceRepository preferenceRepository;
    private final EmailService emailService;
    private final SmsService smsService;
    private final PushService pushService;
    private final DomainEventPublisher eventPublisher;
    
    public NotificationSendingService(NotificationRepository notificationRepository,
                                    NotificationTemplateRepository templateRepository,
                                    NotificationPreferenceRepository preferenceRepository,
                                    EmailService emailService,
                                    SmsService smsService,
                                    PushService pushService,
                                    DomainEventPublisher eventPublisher) {
        this.notificationRepository = notificationRepository;
        this.templateRepository = templateRepository;
        this.preferenceRepository = preferenceRepository;
        this.emailService = emailService;
        this.smsService = smsService;
        this.pushService = pushService;
        this.eventPublisher = eventPublisher;
    }
    
    /**
     * 创建并发送通知
     */
    @Transactional
    public Notification sendNotification(SendNotificationCommand command) {
        // 1. 获取通知偏好
        NotificationPreference preference = preferenceRepository
            .findByPatronId(command.getRecipientId())
            .orElseGet(() -> createDefaultPreference(command.getRecipientId()));
        
        // 2. 检查是否满足发送条件
        if (!shouldSendNotification(command.getNotificationType(), preference)) {
            throw new NotificationNotAuthorizedException(
                "Notification not authorized based on preferences");
        }
        
        // 3. 获取或创建通知
        Notification notification = createOrUpdateNotification(command, preference);
        
        // 4. 立即发送
        sendNotificationNow(notification, preference);
        
        return notification;
    }
    
    /**
     * 调度通知
     */
    @Transactional
    public Notification scheduleNotification(ScheduleNotificationCommand command) {
        NotificationPreference preference = preferenceRepository
            .findByPatronId(command.getRecipientId())
            .orElseGet(() -> createDefaultPreference(command.getRecipientId()));
        
        if (!shouldSendNotification(command.getNotificationType(), preference)) {
            throw new NotificationNotAuthorizedException(
                "Notification not authorized based on preferences");
        }
        
        Notification notification = createOrUpdateNotification(command, preference);
        notification.schedule(command.getScheduledAt());
        
        notificationRepository.save(notification);
        
        return notification;
    }
    
    /**
     * 批量发送通知
     */
    @Transactional
    public List<Notification> sendBulkNotifications(BulkSendCommand command) {
        List<Notification> sentNotifications = new ArrayList<>();
        
        for (RecipientId recipientId : command.getRecipientIds()) {
            try {
                SendNotificationCommand individualCommand = new SendNotificationCommand(
                    command.getNotificationType(),
                    recipientId,
                    command.getSubject(),
                    command.getContent(),
                    command.getVariables()
                );
                
                Notification notification = sendNotification(individualCommand);
                sentNotifications.add(notification);
                
            } catch (Exception e) {
                // 记录失败但继续处理其他通知
                System.err.println("Failed to send notification to " + recipientId + ": " + e.getMessage());
            }
        }
        
        return sentNotifications;
    }
    
    /**
     * 使用模板发送通知
     */
    @Transactional
    public Notification sendTemplateNotification(SendTemplateCommand command) {
        // 1. 获取模板
        NotificationTemplate template = templateRepository
            .findByTemplateCode(command.getTemplateCode())
            .orElseThrow(() -> new TemplateNotFoundException(command.getTemplateCode()));
        
        if (!template.isActive()) {
            throw new TemplateNotActiveException(command.getTemplateCode());
        }
        
        // 2. 渲染通知
        Notification notification = template.render(command.getRecipientId(), command.getVariables());
        
        // 3. 设置优先级
        if (command.getPriority() != null) {
            notification.setPriority(command.getPriority());
        }
        
        // 4. 发送通知
        sendNotificationNow(notification, null);
        
        return notification;
    }
    
    /**
     * 处理待发送通知
     */
    @Transactional
    public void processPendingNotifications() {
        // 1. 获取所有待发送通知
        List<Notification> pendingNotifications = notificationRepository.findPendingNotifications();
        
        for (Notification notification : pendingNotifications) {
            try {
                // 检查是否应该现在发送
                if (notification.getScheduledAt() != null && 
                    notification.getScheduledAt().isAfter(LocalDateTime.now())) {
                    continue; // 还没到发送时间
                }
                
                // 获取偏好设置
                NotificationPreference preference = preferenceRepository
                    .findByPatronId(notification.getRecipientId())
                    .orElse(null);
                
                // 发送通知
                sendNotificationNow(notification, preference);
                
            } catch (Exception e) {
                // 记录失败但继续处理其他通知
                notification.markAsFailed(e.getMessage());
                notificationRepository.save(notification);
            }
        }
    }
    
    /**
     * 处理失败通知的重试
     */
    @Transactional
    public void retryFailedNotifications() {
        List<Notification> failedNotifications = notificationRepository.findFailedNotifications();
        
        for (Notification notification : failedNotifications) {
            try {
                if (notification.canBeRetried()) {
                    notification.retry();
                    notificationRepository.save(notification);
                    
                    // 重新发送
                    sendNotificationNow(notification, null);
                }
            } catch (Exception e) {
                // 重试也失败了
                System.err.println("Retry failed for notification " + notification.getId() + ": " + e.getMessage());
            }
        }
    }
    
    // 私有方法
    private boolean shouldSendNotification(NotificationType notificationType, 
                                         NotificationPreference preference) {
        // 检查类型是否启用
        if (!preference.isTypeEnabled(notificationType)) {
            return false;
        }
        
        // 检查是否在免打扰时段
        if (preference.isInQuietHours()) {
            return false;
        }
        
        return true;
    }
    
    private Notification createOrUpdateNotification(SendNotificationCommand command, 
                                                  NotificationPreference preference) {
        // 如果提供了通知ID，尝试获取现有通知
        if (command.getNotificationId() != null) {
            return notificationRepository.findById(command.getNotificationId())
                .orElseThrow(() -> new NotificationNotFoundException(command.getNotificationId()));
        }
        
        // 创建新通知
        NotificationId notificationId = NotificationId.generate();
        Notification notification = new Notification(
            notificationId,
            command.getNotificationType(),
            command.getRecipientId(),
            command.getSubject(),
            command.getContent()
        );
        
        if (command.getPriority() != null) {
            notification.setPriority(command.getPriority());
        }
        
        // 设置联系信息
        if (preference.getEmailAddress() != null) {
            notification.setRecipientEmail(preference.getEmailAddress());
        }
        if (preference.getPhoneNumber() != null) {
            notification.setRecipientPhone(preference.getPhoneNumber());
        }
        
        return notificationRepository.save(notification);
    }
    
    private void sendNotificationNow(Notification notification, NotificationPreference preference) {
        // 如果没有提供偏好设置，获取默认偏好
        if (preference == null) {
            preference = preferenceRepository.findByPatronId(notification.getRecipientId())
                .orElseGet(() -> createDefaultPreference(notification.getRecipientId()));
        }
        
        // 确定最佳渠道
        NotificationChannel channel = preference.getPreferredChannel(notification.getNotificationType());
        
        try {
            notification.send();
            notificationRepository.save(notification);
            
            // 根据渠道发送
            switch (channel) {
                case EMAIL:
                    sendEmailNotification(notification);
                    break;
                case SMS:
                    sendSmsNotification(notification);
                    break;
                case PUSH:
                    sendPushNotification(notification);
                    break;
                case IN_APP:
                    saveInAppNotification(notification);
                    break;
                default:
                    sendEmailNotification(notification); // 默认邮件
            }
            
        } catch (Exception e) {
            notification.markAsFailed(e.getMessage());
            notificationRepository.save(notification);
            throw e;
        }
    }
    
    private void sendEmailNotification(Notification notification) {
        EmailMessage message = new EmailMessage(
            notification.getRecipientEmail(),
            notification.getSubject(),
            notification.getContent(),
            notification.getId().getValue()
        );
        
        emailService.sendEmail(message);
        notification.markAsDelivered("EMAIL");
        notificationRepository.save(notification);
    }
    
    private void sendSmsNotification(Notification notification) {
        SmsMessage message = new SmsMessage(
            notification.getRecipientPhone(),
            notification.getSubject() + ": " + notification.getContent(),
            notification.getId().getValue()
        );
        
        smsService.sendSms(message);
        notification.markAsDelivered("SMS");
        notificationRepository.save(notification);
    }
    
    private void sendPushNotification(Notification notification) {
        PushMessage message = new PushMessage(
            notification.getRecipientId(),
            notification.getSubject(),
            notification.getContent(),
            notification.getId().getValue()
        );
        
        pushService.sendPush(message);
        notification.markAsDelivered("PUSH");
        notificationRepository.save(notification);
    }
    
    private void saveInAppNotification(Notification notification) {
        notification.markAsDelivered("IN_APP");
        notificationRepository.save(notification);
    }
    
    private NotificationPreference createDefaultPreference(RecipientId recipientId) {
        PreferenceId preferenceId = PreferenceId.generate();
        PatronId patronId = new PatronId(recipientId.getValue());
        
        NotificationPreference preference = new NotificationPreference(preferenceId, patronId);
        return preferenceRepository.save(preference);
    }
}
```

```java
package com.library.notification.infrastructure.service;

import org.springframework.beans.factory.annotation.Value;

import org.springframework.mail.javamail.MimeMessageHelper;
import org.springframework.stereotype.Service;

import javax.mail.internet.MimeMessage;

/**
 * 邮件服务
 */
@Service
public class EmailService {
    
    private final JavaMailSender mailSender;
    
    @Value("${notification.email.from}")
    private String fromEmail;
    
    @Value("${notification.email.from-name}")
    private String fromName;
    
    public EmailService(JavaMailSender mailSender) {
        this.mailSender = mailSender;
    }
    
    public void sendEmail(EmailMessage message) {
        try {
            SimpleMailMessage mailMessage = new SimpleMailMessage();
            mailMessage.setFrom(fromEmail);
            mailMessage.setTo(message.getTo());
            mailMessage.setSubject(message.getSubject());
            mailMessage.setText(message.getContent());
            
            mailSender.send(mailMessage);
            
        } catch (Exception e) {
            throw new EmailSendException("Failed to send email", e);
        }
    }
    
    public void sendHtmlEmail(EmailMessage message) {
        try {
            MimeMessage mimeMessage = mailSender.createMimeMessage();
            MimeMessageHelper helper = new MimeMessageHelper(mimeMessage, true, "UTF-8");
            
            helper.setFrom(fromEmail, fromName);
            helper.setTo(message.getTo());
            helper.setSubject(message.getSubject());
            helper.setText(message.getContent(), true);
            
            mailSender.send(mimeMessage);
            
        } catch (Exception e) {
            throw new EmailSendException("Failed to send HTML email", e);
        }
    }
}
```
